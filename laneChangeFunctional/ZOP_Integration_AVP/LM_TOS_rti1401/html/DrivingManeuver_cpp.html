37{



38	uint8_t t_tem = 0;



39	uint8_t t_count1s = 20;// (uint8_t)(1.0 / 0.05);



40	if (m_lastDMState == m_drivingManeuver)



41	{



42		// keeping



43	}



44	else



45	{



46		m_lastDMState = m_drivingManeuver;



47	}



48	m_drivingManeuverK1 = m_drivingManeuver;



49	m_yawChange1s = 0;



50	m_yawChange2s = 0;



51	for (uint8_t t_i = 0; t_i < t_yaw_st.m_count; t_i++)



52	{



53		if (t_yaw_st.m_firstIndex + t_i >= VEHMOVBUFFER)



54		{



55			t_tem = (t_yaw_st.m_firstIndex + t_i) % VEHMOVBUFFER;



56		}



57		else



58		{



59			t_tem = t_yaw_st.m_firstIndex + t_i;



60		}



61		if (t_i < t_count1s)



62		{



63			m_yawChange1s += t_yaw_st.m_value[t_tem];



64			m_yawChange2s += t_yaw_st.m_value[t_tem];



65		}



66		else



67		{



68			m_yawChange2s += t_yaw_st.m_value[t_tem];



69		}



70	}



71



72



73	StraightStateCalc_V(t_lanemodel_k1_st, t_lanemodel_output_st, t_veh_state, t_vehMovement_st);



74	LeftLaneChangeStateCalc_V(t_lanemodel_k1_st, t_lanemodel_output_st, t_veh_state, t_vehMovement_st);

78	UnknownStateCalc_V(t_lanemodel_k1_st, t_lanemodel_output_st, t_veh_state, t_vehMovement_st);



79



80	if (true == m_straight_b)



81	{



82		m_drivingManeuver = DM_STRAIGHT;



83	}

85	{

239void DrivingManeuver::TurnLeftStateCalc_V(LANE_MODEL_OUTPUT_ST& t_lanemodel_k1_st, LANE_MODEL_OUTPUT_ST& t_lanemodel_output_st, VEHSTATE_ST& t_veh_state, VEHICLE_MOVEMENT_ST& t_vehMovement_st)



240{



241	float t_maxAcc = Interp1(P_Veh2Accl4TurnLOrR_Arr, P_MaxAccl2DiffVx4Trans2TurnLOrR_Arr, 6, t_veh_state.vRef);



242	float t_minDel = Interp1(P_Veh2Decl4TurnLOrR_Arr, P_MaxDecl2DiffVx4Trans2TurnLOrR_Arr, 6, t_veh_state.vRef);



243	float t_MinRawrate = Interp1(P_Veh2Yawrate4TurnLOrR_Arr, P_MinYawrate2DiffVx4Trans2TurnLOrR_Arr, 6, t_veh_state.vRef);



244	float t_MinCurv = Interp1(P_Veh2Curv4TurnLOrR_Arr, P_MinCurv2DiffVx4Trans2TurnLOrR_Arr, 6, t_veh_state.vRef);



245	float t_yaw1s = Interp1(P_Veh2Yaw1s4TurnLOrR_Arr, P_MinYaw1sDiffVx4Trans2TurnLOrR_Arr, 6, t_veh_state.vRef);



246	float t_yaw2s = Interp1(P_Veh2Yaw2s4TurnLOrR_Arr, P_MinYaw2sDiffVx4Trans2TurnLOrR_Arr, 6, t_veh_state.vRef);



247	bool t_lineMarkingCheck_b = false;



248



249	if ((true == t_lanemodel_k1_st.lanes_st[MIDDLE_LANE].available_b)



250		&& (true == t_lanemodel_k1_st.lanes_st[MIDDLE_LANE].segments_st[2].available_b)



251		&& (t_lanemodel_k1_st.lanes_st[MIDDLE_LANE].segments_st[2].coef_arr_f[0] > 0.5)



252		&& (true == t_lanemodel_output_st.lanes_st[MIDDLE_LANE].available_b)



253		&& (true == t_lanemodel_output_st.lanes_st[MIDDLE_LANE].segments_st[2].available_b)



254		&& (t_lanemodel_output_st.lanes_st[MIDDLE_LANE].segments_st[2].coef_arr_f[0] > 0.5)



255		&& (true == t_lanemodel_output_st.lanes_st[MIDDLE_LANE].segments_st[2].detectedByVideo_b)



256		)



257	{



258		t_lineMarkingCheck_b = false;//always within the same lane



259	}



260	else



261	{



262		t_lineMarkingCheck_b = true;



263	}



264	



265	if (DM_TURN_LEFT != m_drivingManeuver)



266	{



267		if ((t_veh_state.vRef < P_MaxVelocity4TurnLeftOrRight_F)// max velocity: 90kph



268			&& (t_veh_state.vRef > P_MinVelocity4TurnLeftOrRight_F)// min velocity: 4kph



269			&& (t_veh_state.aRef < t_maxAcc)



270			&& (t_veh_state.aRef > t_minDel)



271			&& (t_veh_state.YrsF > t_MinRawrate)



272			&& (t_veh_state.Curvature > t_MinCurv)



273			&& (m_yawChange1s > t_yaw1s)



274			&& (m_yawChange2s > t_yaw2s)



275			&& (true == t_lineMarkingCheck_b)



276			)



277		{



278			m_turnLeft_b = true;



279			m_straight_b = false;



280			m_leftLaneChange_b = false;



281			m_rightLaneChange_b = false;



282			m_turnRight_b = false;



283			m_unknown_b = false;



284		}



285	}



286	



287}
