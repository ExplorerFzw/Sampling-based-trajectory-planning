3//***************** Interp1********************************



4// linear interpolation



5// t_xArr:input array - ascending order 



6// t_yArr: input array



7// t_num: the member number of input array



8// t_x: independent variable



9float Interp1(float* t_xArr, float* t_yArr, uint8_t t_num, float t_x)//linear



10{



11	



12	float t_value = 0.0;



13	for (uint8_t t_i = 0; t_i < t_num; t_i++)



14	{



15		if (t_x <= t_xArr[0])



16		{



17			t_value = t_yArr[0];



18			break;



19		}



20		else if (t_x >= t_xArr[t_num - 1])



21		{



22			t_value = t_yArr[t_num - 1];



23			break;



24		}



25		else



26		{



27			if (t_x <= t_xArr[t_i])



28			{



29				t_value = t_yArr[t_i - 1] + (t_x - t_xArr[t_i -1])/(t_xArr[t_i] - t_xArr[t_i - 1])*(t_yArr[t_i] - t_yArr[t_i - 1]);



30				break;



31			}



32		}



33	}

35}



36//***************** Interp2********************************



37// linear interpolation



38// t_xArr:input array - ascending order 



39// t_yArr: input array - ascending order 



40// t_zArr: input array



41// t_num: the member number of input array



42// t_x: independent variable



43// t_y: independent variable



44float Interp2(float* t_xArr, float* t_yArr, float* t_zArr, uint8_t t_num, float t_x, float t_y)//linear



45{



46	float t_value = 0.0, t_xRatio = 0,t_yRatio = 0;



47	uint8_t t_xIndex = 0, t_yIndex = 0;



48	if ((t_x <= t_xArr[0]) && (t_y <= t_yArr[0]))



49	{



50		t_value = t_zArr[0];



51	}



52	else if ((t_x >= t_xArr[t_num - 1]) && (t_y >= t_yArr[t_num - 1]))



53	{



54		t_value = t_zArr[t_num* t_num - 1];



55	}



56	else



57	{



58		float t_tem = 0, t_temX1 = 0, t_temX2 = 0, t_temZ0 = 0, t_temZ1 = 0, t_temZ2 = 0, t_temZ3 = 0;



59		if (t_x <= t_xArr[0])



60		{



61			t_xIndex = 0;



62			t_xRatio = 0.0;



63		}



64		else if (t_x >= t_xArr[t_num - 1])



65		{



66			t_xIndex = t_num - 2;



67			t_xRatio = 1;



68		}



69		else



70		{



71			for (uint8_t t_i = 1; t_i < t_num; t_i++)



72			{



73				if (t_x <= t_xArr[t_i])



74				{



75					t_xIndex = t_i - 1;



76					t_xRatio = (t_x - t_xArr[t_i - 1]) / (t_xArr[t_i] - t_xArr[t_i - 1]);



77					break;



78				}



79			}



80		}



81



82



83		if (t_y <= t_yArr[0])



84		{



85			t_yIndex = 0;



86			t_yRatio = 0.0;



87		}



88		else if (t_y >= t_yArr[t_num - 1])



89		{



90			t_yIndex = t_num - 2;



91			t_yRatio = 1.0;



92		}



93		else



94		{



95			for (uint8_t t_i = 1; t_i < t_num; t_i++)



96			{



97				if (t_y <= t_yArr[t_i])



98				{



99					t_yIndex = t_i - 1;



100					t_yRatio = (t_y - t_yArr[t_i - 1]) / (t_yArr[t_i] - t_yArr[t_i - 1]);



101					break;



102				}



103			}



104		}



105



106		



107		t_temZ0 = t_zArr[t_xIndex * t_num + t_yIndex];



108		t_temZ1 = t_zArr[(t_xIndex+1) * t_num + t_yIndex];



109		t_temZ2 = t_zArr[t_xIndex  * t_num + t_yIndex + 1];



110		t_temZ3 = t_zArr[(t_xIndex + 1) * t_num + t_yIndex + 1];



111		t_temX1 = t_temZ0 + t_xRatio * (t_temZ1 - t_temZ0);



112		t_temX2 = t_temZ2 + t_xRatio * (t_temZ3 - t_temZ2);



113		t_tem = t_temX1 + t_yRatio*(t_temX2 - t_temX1);



114		t_value = t_tem;



115	}



116	



117	return t_value;



118}

120//***************** polinomial fit********************************



121// polinomial : y=c0 + c1*x + c2*x*x + c3*x*x*x



122// t_pointNum: the number of points



123// points[]: include the x and y 



124// COEFF: a struct is used to save the coefficient: c0, c1, c2,c3



125



126void polfit3(POINT_X points[], int t_pointNum, COEFF& t_coeff)//最小二乘 三次多项式拟合 y=c0 + c1*x + c2*x*x + c3*x*x*x  



127{



128



129



130	//-----------------basic method---------------------------



131	//n c0     + sumX c1   + sumXX c2   +  sumXXX c3   = sumY



132	//sumX c0  + sumXX c1  + sumXXX c2  +  sumXXXX c3  = sumXY



133	//sumXX c0 + sumXXX c1 + sumXXXX c2 +  sumXXXXX c3 = sumXXY



134	//sumXXX c0+sumXXXX c1 + sumXXXXX c2+  sumXXXXXX c3 =sumXXXY



135



136	long double sumX = 0, sumY = 0, sumXX = 0, sumXXX = 0, sumXXXX = 0, sumXXXXX = 0, sumXXXXXX = 0, sumXXXXXXX = 0, sumXY = 0, sumXXY = 0, sumXXXY = 0;//建立方程组



137	for (int t_i = 0; t_i < t_pointNum; t_i++)



138	{



139		sumX = sumX + points[t_i].x;



140		sumY = sumY + points[t_i].y;



141		sumXX = sumXX + points[t_i].x * points[t_i].x;



142		sumXXX = sumXXX + points[t_i].x * points[t_i].x * points[t_i].x;



143		sumXXXX = sumXXXX + points[t_i].x * points[t_i].x * points[t_i].x * points[t_i].x;



144		sumXXXXX = sumXXXXX + points[t_i].x * points[t_i].x * points[t_i].x * points[t_i].x * points[t_i].x;//



145		sumXXXXXX = sumXXXXXX + points[t_i].x * points[t_i].x * points[t_i].x * points[t_i].x * points[t_i].x * points[t_i].x;



146		//	sumXXXXXXX = sumXXXXXXX + points[t_i].x*points[t_i].x*points[t_i].x*points[t_i].x*points[t_i].x*points[t_i].x*points[t_i].x;



147		sumXY = sumXY + points[t_i].x * points[t_i].y;



148		sumXXY = sumXXY + points[t_i].x * points[t_i].x * points[t_i].y;



149		sumXXXY = sumXXXY + points[t_i].x * points[t_i].x * points[t_i].x * points[t_i].y;



150



151	}



152



153	double  a11 = t_pointNum; double  a12 = sumX;    double  a13 = sumXX;    double  a14 = sumXXX;    double  b1 = sumY;      //构造矩阵   求解方程系数   a11 第一行第一列



154	double  a21 = sumX;		  double  a22 = sumXX;   double  a23 = sumXXX;   double  a24 = sumXXXX;   double  b2 = sumXY;



155	double  a31 = sumXX;      double  a32 = sumXXX;  double  a33 = sumXXXX;  double  a34 = sumXXXXX;  double  b3 = sumXXY;



156	double  a41 = sumXXX;     double  a42 = sumXXXX; double  a43 = sumXXXXX; double  a44 = sumXXXXXX; double  b4 = sumXXXY;



157	//double a11 = 2;      double  a12 = 4;   double  a13 = -2;   double  b1 = 2;      //构造矩阵   求解方程系数   a11 第一行第一列



158	//double  a21 = 1;  double  a22 = -3;  double  a23 = -3;  double  b2 = -1;



159	//double  a31 = 4; double  a32 = 2; double  a33 = 2; double  b3 = 3;



160



161	//R2=R2-(a21/a11)*R1     ------------------------------------------------------------ - 进行消元    高斯消元法  第一例消元



162	double a21x, a22x, a23x, a24x, b2x;



163	a21x = a21 - a21 / a11 * a11; a22x = a22 - a21 / a11 * a12; a23x = a23 - a21 / a11 * a13; a24x = a24 - a21 / a11 * a14; b2x = b2 - a21 / a11 * b1;



164	//R3=R3-(a31/a11)*R1



165	double a31x, a32x, a33x, a34x, b3x;



166	a31x = a31 - a31 / a11 * a11; a32x = a32 - a31 / a11 * a12; a33x = a33 - a31 / a11 * a13; a34x = a34 - a31 / a11 * a14; b3x = b3 - a31 / a11 * b1;



167	//R4=R4-(a41/a11)*R1



168	double a41x, a42x, a43x, a44x, b4x;

172	//R3=R3-(a32/a22)*R2



173	double a32xx, a33xx, a34xx, b3xx;

175	//R4=R4-(a42/a22)*R2



176	double a42xx, a43xx, a44xx, b4xx;



177	a42xx = a42x - a42x / a22x * a22x; a43xx = a43x - a42x / a22x * a23x; a44xx = a44x - a42x / a22x * a24x; b4xx = b4x - a42x / a22x * b2x;
