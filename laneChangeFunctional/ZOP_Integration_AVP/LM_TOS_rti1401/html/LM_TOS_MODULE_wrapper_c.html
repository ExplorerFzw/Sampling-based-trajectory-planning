97/* %%%-SFUNWIZ_wrapper_Outputs_Changes_BEGIN --- EDIT HERE TO _END */

108ats_to_st.vx = acc_to_In->vx;

112ats_to_st.axAbs = acc_to_In->axAbs;



113ats_to_st.vxAbs = acc_to_In->vxAbs;

116// calc



117LM_TOS_RUN();

131}

133{



134    rolling_count_output[0] = rolling_count_output[0]+1;



135}
