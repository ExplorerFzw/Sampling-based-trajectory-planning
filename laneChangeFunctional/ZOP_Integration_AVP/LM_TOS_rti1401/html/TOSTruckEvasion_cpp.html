31void TOS_TRUCK_EVASION::Calc(FUSION_OBJ_OUTPUT_ST& fus_obj_output_st,TOS_OBJ_PROB_WITHIN_LANE& t_tos_obj_within_lane_st, PROB_EGO_VEHICLE_WITHIN_LANE_ST& t_probEgoVehicleWithinLane_st)



32{



33	LeftSide_Calc(fus_obj_output_st, t_tos_obj_within_lane_st, t_probEgoVehicleWithinLane_st);



34	RightSide_Calc( fus_obj_output_st,t_tos_obj_within_lane_st, t_probEgoVehicleWithinLane_st);



35	if ((true == m_tar_obj_l_st.available_b) &&



36		(true == m_tar_obj_r_st.available_b)



37		)



38	{



39		m_TE_Flag = TE_LEFT_AND_RIGHT;



40	}



41	else if (true == m_tar_obj_l_st.available_b)



42	{



43		m_TE_Flag = TE_LEFT;



44	}



45	else if (true == m_tar_obj_r_st.available_b)



46	{



47		m_TE_Flag = TE_RIGHT;



48	}



49	else



50	{



51		m_TE_Flag = TE_INVALID;



52	}



53}



54void TOS_TRUCK_EVASION::LeftSide_Calc(FUSION_OBJ_OUTPUT_ST& fus_obj_output_st, TOS_OBJ_PROB_WITHIN_LANE& t_tos_obj_within_lane_st, PROB_EGO_VEHICLE_WITHIN_LANE_ST& t_probEgoVehicleWithinLane_st)



55{



56	//req:



57	//0:truck



58	//1:within left lane



59	//2:-20 << dx < 100



60	//2: TTC < 10s



61	uint8_t t_index = 0;



62	bool t_nofind = true;



63	for (uint8_t t_i = 0; (t_i < FUS_MAX_OBJ_NUM)&&(true == t_nofind); t_i++)



64	{



65		t_index = t_tos_obj_within_lane_st.m_obj_list_index_un8[t_i];



66		if (true == m_tar_obj_l_st.available_b)



67		{



68			if ((t_index < FUS_MAX_OBJ_NUM) &&



69				(LEFT_LANE == t_tos_obj_within_lane_st.m_withinLaneId[t_index]) &&



70				(uint8_t(TRUCK) == fus_obj_output_st.objects[t_index].obj_information.type) &&



71				(fus_obj_output_st.objects[t_index].obj_estimate.longPosition < 100 + 10) &&



72				(fus_obj_output_st.objects[t_index].obj_estimate.longPosition > -(20 + 5)) &&



73				(t_tos_obj_within_lane_st.m_tarObj_arr_st[t_index].TTC_f < 10)



74				)



75			{



76				t_nofind = false;



77



78			}



79		}



80		else



81		{



82			if ((t_index < FUS_MAX_OBJ_NUM) &&



83				(LEFT_LANE == t_tos_obj_within_lane_st.m_withinLaneId[t_index]) &&



84				(uint8_t(TRUCK) == fus_obj_output_st.objects[t_index].obj_information.type) &&



85				(fus_obj_output_st.objects[t_index].obj_estimate.longPosition < 100) &&



86				(fus_obj_output_st.objects[t_index].obj_estimate.longPosition > -20) &&



87				(t_tos_obj_within_lane_st.m_tarObj_arr_st[t_index].TTC_f < 9)



88				)



89			{



90				t_nofind = false;



91



92			}



93		}



94		



95	}



96



97	if (false == t_nofind)



98	{



99		m_tar_obj_l_st.available_b = true;



100		m_tar_obj_l_st.dx = fus_obj_output_st.objects[t_index].obj_estimate.longPosition;



101		m_tar_obj_l_st.dy = fus_obj_output_st.objects[t_index].obj_estimate.latPosition;



102		m_tar_obj_l_st.id = fus_obj_output_st.objects[t_index].obj_property.id;



103		m_tar_obj_l_st.index = t_index;



104		m_tar_obj_l_st.vx = fus_obj_output_st.objects[t_index].obj_estimate.longAbsoluteVelocity;



105		m_tar_obj_l_st.vy = fus_obj_output_st.objects[t_index].obj_estimate.latAbsoluteVelocity;



106	}



107	else



108	{



109		m_tar_obj_l_st.available_b = false;



110		m_tar_obj_l_st.dx = MAX_F;



111		m_tar_obj_l_st.dy = MAX_F;



112		m_tar_obj_l_st.id = MAX_UN8;



113		m_tar_obj_l_st.index = MAX_UN8;



114		m_tar_obj_l_st.vx = MAX_F;



115		m_tar_obj_l_st.vy = MAX_F;



116	}



117



118	



119



120}
