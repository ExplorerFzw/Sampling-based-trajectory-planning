21void CoordinationTransformation::VehMovCalc(VEHSTATE_ST& t_vehState_st)



22{



23    float t_vEgo_f;



24    float t_tDtCycle_f;



25    float t_psiDt_f;



26    float t_kapCurvTraj_f;



27    float t_psi_f;



28    float t_SinOfPsi_f;



29    float t_CosOfPsi_f;



30    float t_dDeltaX_f;



31    float t_dDeltaY_f;



32    float t_dEgoCurveRadius_f;



33    float t_dDeltaXTemp_f;



34    float t_SinOf05Psi_f;



35    float t_dDeltaYTemp_f;



36    float t_SlipAngleRearAxle_f;



37



38    /* calc ego motion */



39    t_vEgo_f = t_vehState_st.vRef;



40    t_tDtCycle_f = GetCycleTime_s();



41    t_psiDt_f = t_vehState_st.YrsF;



42    t_kapCurvTraj_f = t_vehState_st.Curvature;



43



44    if (t_kapCurvTraj_f < 1/3000)



45    {



46        t_psi_f = 0.0f;



47        t_SinOfPsi_f = sinf(t_psi_f);



48        t_CosOfPsi_f = cosf(t_psi_f);



49



50        t_dDeltaX_f = (t_vEgo_f * t_tDtCycle_f) * t_CosOfPsi_f;



51        t_dDeltaY_f = (t_vEgo_f * t_tDtCycle_f) * t_SinOfPsi_f;



52    }



53    else



54    {



55        



56        if ((fabsf(t_kapCurvTraj_f) > 1/20 )//min radius



57            && (fabsf(t_vEgo_f) > 1.0))



58        {



59            float t_kapCurvTrajFromPsiDt_f;



60



61            t_kapCurvTrajFromPsiDt_f = t_psiDt_f / t_vEgo_f;



62



63            if (fabsf(t_kapCurvTrajFromPsiDt_f) > fabsf(t_kapCurvTraj_f))



64            {



65                t_kapCurvTraj_f = t_kapCurvTrajFromPsiDt_f;



66            }



67        }



68



69        t_dEgoCurveRadius_f = (1.0f / t_kapCurvTraj_f);



70        t_psi_f = (t_kapCurvTraj_f * t_vEgo_f) * t_tDtCycle_f;



71



72        t_SinOfPsi_f = sinf(t_psi_f);



73        t_CosOfPsi_f = cosf(t_psi_f);



74



75        t_dDeltaXTemp_f = t_dEgoCurveRadius_f * t_SinOfPsi_f;



76        /* numerical more robust version: */



77        t_SinOf05Psi_f = sinf(t_psi_f / 2.0f);



78        t_dDeltaYTemp_f = t_dEgoCurveRadius_f * (2 * (t_SinOf05Psi_f * t_SinOf05Psi_f));



79



80        t_SlipAngleRearAxle_f = 0;//TODO



81



82        // regard slip angle for calculation of ego postion



83        t_dDeltaX_f = (cosf(t_SlipAngleRearAxle_f) * t_dDeltaXTemp_f) - (sinf(t_SlipAngleRearAxle_f) * t_dDeltaYTemp_f);



84        t_dDeltaY_f = (sinf(t_SlipAngleRearAxle_f) * t_dDeltaXTemp_f) + (cosf(t_SlipAngleRearAxle_f) * t_dDeltaYTemp_f);



85



86    }



87



88    m_vehMovement_st.deltaDx = t_dDeltaX_f;



89    m_vehMovement_st.deltaDy = t_dDeltaY_f;



90    m_vehMovement_st.deltaDyaw = t_psi_f;



91}

93void CoordinationTransformation::CoorTransmAsEgoMov(float &t_dx, float &t_dy)



94{



95    float t_dX_f;



96    float t_dY_f;



97



98    t_dX_f = t_dx - m_vehMovement_st.deltaDx;



99    t_dY_f = t_dy - m_vehMovement_st.deltaDy;



100    t_dx = t_dX_f * cosf(m_vehMovement_st.deltaDyaw) + t_dY_f * sinf(m_vehMovement_st.deltaDyaw);



101    t_dy = -t_dX_f * sinf(m_vehMovement_st.deltaDyaw) + t_dY_f * cosf(m_vehMovement_st.deltaDyaw);   



102}
