34void TOS_TRUCK_EVASION::LeftSide_Calc(FUSION_OBJ_OUTPUT_ST& fus_obj_output_st, TOS_OBJ_PROB_WITHIN_LANE& t_tos_obj_within_lane_st, PROB_EGO_VEHICLE_WITHIN_LANE_ST& t_probEgoVehicleWithinLane_st)



35{



36	//req:



37	//0:truck



38	//1:within left lane



39	//2:-20 << dx < 100



40	//2: TTC < 10s



41	uint8_t t_index = 0;



42	bool t_nofind = true;



43	for (uint8_t t_i = 0; (t_i < FUS_MAX_OBJ_NUM)&&(true == t_nofind); t_i++)



44	{



45		t_index = t_tos_obj_within_lane_st.m_obj_list_index_un8[t_i];



46		if (true == m_tar_obj_l_st.available_b)



47		{



48			if ((t_index < FUS_MAX_OBJ_NUM) &&



49				(LEFT_LANE == t_tos_obj_within_lane_st.m_withinLaneId[t_index]) &&



50				(uint8_t(TRUCK) == fus_obj_output_st.objects[t_index].obj_information.type) &&



51				(fus_obj_output_st.objects[t_index].obj_estimate.longPosition < 100 + 10) &&



52				(fus_obj_output_st.objects[t_index].obj_estimate.longPosition > -(20 + 5)) &&



53				(t_tos_obj_within_lane_st.m_tarObj_arr_st[t_index].TTC_f < 10)



54				)



55			{



56				t_nofind = false;



57



58			}



59		}



60		else



61		{



62			if ((t_index < FUS_MAX_OBJ_NUM) &&



63				(LEFT_LANE == t_tos_obj_within_lane_st.m_withinLaneId[t_index]) &&



64				(uint8_t(TRUCK) == fus_obj_output_st.objects[t_index].obj_information.type) &&



65				(fus_obj_output_st.objects[t_index].obj_estimate.longPosition < 100) &&



66				(fus_obj_output_st.objects[t_index].obj_estimate.longPosition > -20) &&



67				(t_tos_obj_within_lane_st.m_tarObj_arr_st[t_index].TTC_f < 9)



68				)



69			{



70				t_nofind = false;



71



72			}



73		}



74		



75	}



76



77	if (false == t_nofind)



78	{



79		m_tar_obj_l_st.available_b = true;



80		m_tar_obj_l_st.dx = fus_obj_output_st.objects[t_index].obj_estimate.longPosition;



81		m_tar_obj_l_st.dy = fus_obj_output_st.objects[t_index].obj_estimate.latPosition;



82		m_tar_obj_l_st.id = fus_obj_output_st.objects[t_index].obj_property.id;



83		m_tar_obj_l_st.index = t_index;



84		m_tar_obj_l_st.vx = fus_obj_output_st.objects[t_index].obj_estimate.longAbsoluteVelocity;



85		m_tar_obj_l_st.vy = fus_obj_output_st.objects[t_index].obj_estimate.latAbsoluteVelocity;



86	}



87	else



88	{



89		m_tar_obj_l_st.available_b = false;



90		m_tar_obj_l_st.dx = MAX_F;



91		m_tar_obj_l_st.dy = MAX_F;



92		m_tar_obj_l_st.id = MAX_UN8;



93		m_tar_obj_l_st.index = MAX_UN8;



94		m_tar_obj_l_st.vx = MAX_F;



95		m_tar_obj_l_st.vy = MAX_F;



96	}



97



98	



99



100}
