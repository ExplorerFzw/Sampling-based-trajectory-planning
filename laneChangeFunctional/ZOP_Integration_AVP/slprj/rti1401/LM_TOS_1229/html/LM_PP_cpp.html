28void LM_PP::BC_Check_Calc(FUSION_OBJ_OUTPUT_ST& obj_data_st, VEHSTATE_ST& veh_state, LANE_MODEL_OUTPUT_ST& lane_medel)



29{



30	float t_front_range_f = 130.0;



31	float t_rear_range_f = -90.0;



32	float t_left_range_f = 9.0;



33	float t_right_range_f = -9.0;



34



35	for (uint8_t t_i = 0; t_i < FUS_MAX_OBJ_NUM; t_i++)



36	{



37		bool t_movingPatternCheck_b = false;



38		if ((PRECEDING == obj_data_st.objects[t_i].obj_information.motionPatternCurrent) ||



39			(ONCOMING == obj_data_st.objects[t_i].obj_information.motionPatternCurrent) ||



40			(STOP == obj_data_st.objects[t_i].obj_information.motionPatternCurrent)



41			)



42		{



43			t_movingPatternCheck_b = true;



44		}



45		else if ((STATIONARY == obj_data_st.objects[t_i].obj_information.motionPatternCurrent) &&



46			((RADARVISIONFUSION == obj_data_st.objects[t_i].obj_property.fusionSource)||



47				(VISIONONLY == obj_data_st.objects[t_i].obj_property.fusionSource)



48				)



49			)



50		{



51			t_movingPatternCheck_b = true;



52		}



53		else



54		{



55			t_movingPatternCheck_b = false;//keeping



56		}



57



58		if ((veh_state.vRef >= -1.0) &&



59			(true == obj_data_st.objects[t_i].obj_information.valid)&&



60			(fabsf(obj_data_st.objects[t_i].obj_estimate.latAbsoluteVelocity) < 3.0) &&



61			(obj_data_st.objects[t_i].obj_estimate.longAbsoluteVelocity > -1.0) &&



62			//(obj_data_st.objects[t_i].obj_estimate.latPosition < t_left_range_f)&&



63			//(obj_data_st.objects[t_i].obj_estimate.latPosition > t_right_range_f)&



64			(obj_data_st.objects[t_i].obj_estimate.longPosition < t_front_range_f) &&



65			(obj_data_st.objects[t_i].obj_estimate.longPosition > t_rear_range_f)&&



66			(t_movingPatternCheck_b)



67			)



68		{



69			if (true == m_obj_list_by_bc_check_st.valid_b[t_i])



70			{



71				if (m_obj_list_by_bc_check_st.trackTime[t_i] < MAX_UN16)



72				{



73					m_obj_list_by_bc_check_st.trackTime[t_i] += 1;



74				}



75				else



76				{



77					m_obj_list_by_bc_check_st.trackTime[t_i] = MAX_UN16;



78				}



79			}



80			else



81			{



82				m_obj_list_by_bc_check_st.trackTime[t_i] = 1;



83				m_obj_list_by_bc_check_st.id[t_i] = obj_data_st.objects[t_i].obj_property.id;



84				m_obj_list_by_bc_check_st.valid_b[t_i] = true;



85			}



86		}



87		else



88		{



89			m_obj_list_by_bc_check_st.id[t_i] = MAX_UN8;



90			m_obj_list_by_bc_check_st.valid_b[t_i] = false;



91			m_obj_list_by_bc_check_st.trackTime[t_i] = 0;



92		}



93



94	}



95



96}

107void LM_PP::LM_General_Limit_Calc(VEHSTATE_ST& veh_state)



108{



109	float t_egoVehVel_f = veh_state.vRef;



110	float t_segmentLength_f = 1.27 * t_egoVehVel_f + 14.6;



111	float t_minLength_f = 13.3;



112	float t_maxLength_f = 30.0;



113	float t_factor_rear = 1.5;



114	float t_factor_front = 1;



115



116	



117	t_segmentLength_f = MIN_TP(t_segmentLength_f, t_maxLength_f);



118	t_segmentLength_f = MAX_TP(t_segmentLength_f, t_minLength_f);



119	for (uint8_t t_i = 0; t_i < LM_MAX_NUMBER_OF_COURSE_LIMITS; t_i++)



120	{



121		if (t_i >= uint8_t(2))



122		{



123			m_lm_general_limit_st.m_d_lower_limit_f[t_i] = m_lm_general_limit_st.m_d_upper_limit_f[t_i -1];



124			float t_upper = m_lm_general_limit_st.m_d_lower_limit_f[t_i] + t_segmentLength_f* t_factor_front;



125			float t_tem = LM_MAX_FRONT_RANGE;



126			m_lm_general_limit_st.m_d_upper_limit_f[t_i] = MIN_TP(t_tem, t_upper);



127		}



128		else if (uint8_t(0) == t_i)



129		{



130			m_lm_general_limit_st.m_d_upper_limit_f[1] = 0;



131			m_lm_general_limit_st.m_d_lower_limit_f[1] = -t_segmentLength_f* t_factor_rear;



132			m_lm_general_limit_st.m_d_upper_limit_f[0] = m_lm_general_limit_st.m_d_lower_limit_f[1];



133			float t_upper = m_lm_general_limit_st.m_d_upper_limit_f[0] - t_segmentLength_f* t_factor_rear;



134			float t_tem = LM_MAX_REAR_RANGE;



135			m_lm_general_limit_st.m_d_lower_limit_f[0] = MAX_TP(t_tem, t_upper);



136		}



137		else



138		{



139			//nothing



140		}



141	}



142}
