10void HMI::Init()



11{



12	hmi_display_obj.object_count = NULL_UN8;



13	for (uint8_t t_i = 0; t_i < MAX_OBJ_NUM_4_HMI; t_i++)



14	{



15		hmi_display_obj.objects[t_i].dx = MAX_F;



16		hmi_display_obj.objects[t_i].dy = MAX_F;



17		hmi_display_obj.objects[t_i].object_ID = MAX_UN8;



18		hmi_display_obj.objects[t_i].obj_type = CLASS_DEFAULT;



19		hmi_display_obj.objects[t_i].obj_attr = CLASS_SPE_DEFAULT;



20



21	}



22}

24void HMI::Calc(LANE_MODEL_OUTPUT_ST& lane_medel,FUSION_OBJ_OUTPUT_ST& fus_output_st,TOS_OBJ_PROB_WITHIN_LANE& t_tos_obj_within_lane_st,TRUCK_EVASION_TAR_OBJ& t_tar_obj_l_st, TRUCK_EVASION_TAR_OBJ& t_tar_obj_r_st, ATS_TO_ST& t_ats_po_st)



25{



26	uint8_t t_HMI_MAX_Obj = MAX_OBJ_NUM_4_HMI;



27	uint8_t t_numEachLane = MAX_UN8, t_totalNumObj = MAX_UN8, t_index = MAX_UN8;



28	for (uint8_t t_lane = LEFT_LANE; (t_lane <= RIGHT_LANE)&&(t_totalNumObj < t_HMI_MAX_Obj); t_lane++)



29	{



30		t_numEachLane = lane_medel.lanes_st[t_lane].objectNum_un8;



31		for (uint8_t t_i = 0; t_i < t_numEachLane; t_i++)



32		{



33			t_index = lane_medel.lanes_st[t_lane].ObjectList_st[t_i].Index_un8;



34			hmi_display_obj.objects[t_totalNumObj].object_ID = lane_medel.lanes_st[t_lane].ObjectList_st[t_i].ID_un8 ;



35			hmi_display_obj.objects[t_totalNumObj].dx = lane_medel.lanes_st[t_lane].ObjectList_st[t_i].dx;



36			hmi_display_obj.objects[t_totalNumObj].dy = lane_medel.lanes_st[t_lane].ObjectList_st[t_i].dy;



37			if (uint8_t(PEDESTRIAN) == fus_output_st.objects[t_index].obj_information.type)



38			{



39				hmi_display_obj.objects[t_totalNumObj].obj_type = CLASS_PEDESTRIAN;



40			}



41			else if (uint8_t(CYCLIST) == fus_output_st.objects[t_index].obj_information.type)



42			{



43				hmi_display_obj.objects[t_totalNumObj].obj_type = CLASS_BIKE;



44			}



45			else if ((uint8_t(SMALL_MEDIUM_CAR) == fus_output_st.objects[t_index].obj_information.type)||



46				(uint8_t(TINY_CAR) == fus_output_st.objects[t_index].obj_information.type)||



47				(uint8_t(UNCERTAIN_VEHICLE) == fus_output_st.objects[t_index].obj_information.type) ||



48				(uint8_t(BUS) == fus_output_st.objects[t_index].obj_information.type) ||



49				(uint8_t(AMBULANCE) == fus_output_st.objects[t_index].obj_information.type) ||



50				(uint8_t(TRI_MOTOR) == fus_output_st.objects[t_index].obj_information.type)



51				)



52			{



53				hmi_display_obj.objects[t_totalNumObj].obj_type = CLASS_CAR;



54			}



55			else if (uint8_t(TRUCK) == fus_output_st.objects[t_index].obj_information.type)



56			{



57				hmi_display_obj.objects[t_totalNumObj].obj_type = CLASS_TRUCK;



58			}



59			else



60			{



61				hmi_display_obj.objects[t_totalNumObj].obj_type = CLASS_DEFAULT;



62			}



63			if (t_ats_po_st.index == t_index)



64			{



65				hmi_display_obj.objects[t_totalNumObj].obj_attr = CLASS_ACC_FRONT_CAR;



66			}



67			else if (lane_medel.lanes_st[t_lane].ObjectList_st[t_i].probCutIn_f > 0.6)



68			{



69				hmi_display_obj.objects[t_totalNumObj].obj_attr = CLASS_CUT_IN;



70			}



71			else if ((uint8_t(TRUCK) == fus_output_st.objects[t_index].obj_information.type) &&



72				((t_i == t_tar_obj_r_st.index) || (t_i == t_tar_obj_l_st.index))



73				)



74			{



75				hmi_display_obj.objects[t_totalNumObj].obj_attr = CLASS_TRUCK_EVASION;



76			}



77			else if ((LEFT_LANE == t_tos_obj_within_lane_st.m_withinLaneId[t_i])||



78				(RIGHT_LANE == t_tos_obj_within_lane_st.m_withinLaneId[t_i]))



79			{



80				hmi_display_obj.objects[t_totalNumObj].obj_attr = CLASS_OBSTACLE_ON_ADJACENT_LANE;



81			}



82			else



83			{



84				hmi_display_obj.objects[t_totalNumObj].obj_attr = CLASS_SPE_DEFAULT;



85			}



86			t_totalNumObj++;



87



88		}



89	}



90



91	hmi_display_obj.object_count = t_totalNumObj;



92}
