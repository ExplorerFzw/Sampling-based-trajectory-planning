99void VSL::findTheBestLine(LM_GENERAL_LIMIT_ST& inputLimits_pst, VISION_LANE_INFO_ST& vision_lane_info_st, VEHSTATE_ST& veh_state, VEHICLE_MOVEMENT_ST& t_vehMovement_st)



100{



101	uint8_t const t_line_num = VSL_MAX_LINE_NUM;



102	uint8_t t_line_length_order[t_line_num];



103	float t_length_f = 0, t_tem = 0, t_line_length[t_line_num];



104	for (uint8_t t_i = 0; t_i < t_line_num; t_i++)



105	{



106		t_line_length_order[t_i] = MAX_UN8;



107		t_line_length[t_i] = 0;



108	}



109



110	for (uint8_t t_i = 0; t_i < t_line_num; t_i++)



111	{



112		if ((MEASUREMENT == vision_lane_info_st.lines_st_arr[t_i].traceStatus) ||



113			(PREDICTED == vision_lane_info_st.lines_st_arr[t_i].traceStatus))



114		{



115			t_line_length[t_i] = vision_lane_info_st.lines_st_arr[t_i].curFunction.longDistanceToend;



116		}



117		else



118		{



119			t_line_length[t_i] = 0;



120		}



121	}



122	float t_tem_length = 0;



123	// order



124	for (uint8_t t_i = 0; t_i < t_line_num - 1; t_i++)



125	{



126		for (uint8_t t_j = t_i + 1; t_j < t_line_num; t_j++)



127		{



128			if ((t_line_length[t_i] < t_line_length[t_j]) && (t_tem_length + FLOAT_EPSILON < t_line_length[t_j]))



129			{



130				t_line_length_order[t_i] = t_j;



131				t_line_length_order[t_j] = t_i;



132				t_tem_length = t_line_length[t_j];



133			}



134			else if (t_line_length[t_j] <= FLOAT_EPSILON)



135			{



136				t_line_length_order[t_j] = MAX_UN8;



137			}



138		}



139	}



140	// find the best line with higher length and higher quality



141	bool t_no_find_b = true;



142	bool t_longer_length_with_low_quality_b = false;



143	m_best_line_info_st.m_line_id_un8 = MAX_UN8;// init



144	for (uint8_t t_i = 0; (t_i < t_line_num) && t_no_find_b; t_i++)



145	{



146		uint8_t t_tem = t_line_length_order[t_i];



147		if ((t_line_length_order[t_i] < 4) &&



148			((VISION_LINE_MIDDLE_HIGH == vision_lane_info_st.lines_st_arr[t_tem].quality) ||



149				(VISION_LINE_HIGH == vision_lane_info_st.lines_st_arr[t_tem].quality))



150			)



151		{



152			m_best_line_info_st.m_line_id_un8 = t_tem;



153			m_best_line_info_st.m_length_f = vision_lane_info_st.lines_st_arr[t_tem].curFunction.longDistanceToend;



154			m_best_line_info_st.m_coeff_f[0] = vision_lane_info_st.lines_st_arr[t_tem].curFunction.latDistanceZeroOrderCoeff;



155			m_best_line_info_st.m_coeff_f[1] = vision_lane_info_st.lines_st_arr[t_tem].curFunction.latDistanceFirstOrderCoeff;



156			m_best_line_info_st.m_coeff_f[2] = vision_lane_info_st.lines_st_arr[t_tem].curFunction.latDistanceSecondOrderCoeff;



157			m_best_line_info_st.m_coeff_f[3] = vision_lane_info_st.lines_st_arr[t_tem].curFunction.latDistanceThirdOrderCoeff;



158			t_no_find_b = false;



159		}



160		else//case: all the line with low quality



161		{



162			if ((0 == t_i) && (MAX_UN8 != t_line_length_order[0]))



163			{



164				m_best_line_info_st.m_line_id_un8 = t_tem;



165				m_best_line_info_st.m_length_f = vision_lane_info_st.lines_st_arr[t_tem].curFunction.longDistanceToend;



166				m_best_line_info_st.m_coeff_f[0] = vision_lane_info_st.lines_st_arr[t_tem].curFunction.latDistanceZeroOrderCoeff;



167				m_best_line_info_st.m_coeff_f[1] = vision_lane_info_st.lines_st_arr[t_tem].curFunction.latDistanceFirstOrderCoeff;



168				m_best_line_info_st.m_coeff_f[2] = vision_lane_info_st.lines_st_arr[t_tem].curFunction.latDistanceSecondOrderCoeff;



169				m_best_line_info_st.m_coeff_f[3] = vision_lane_info_st.lines_st_arr[t_tem].curFunction.latDistanceThirdOrderCoeff;



170			}



171			else



172			{



173				//nothing



174			}



175



176		}



177	}



178



179	// no line available



180



181	//************************calc lane width********************************



182	float t_lane_width_f = 0;



183	float t_max_lane_width = LM_MAX_LANE_WIDTH;



184	float t_min_lane_width = LM_MIN_LANE_WIDTH;



185	//for (uint8_t t_i = 0; t_i < VSL_MAX_LANE_NUM; t_i++)



186	{



187



188		//middle lane width



189		if ((vision_lane_info_st.hostLaneWidth > 2.1) && (vision_lane_info_st.hostLaneWidth < 5))



190		{



191			// ego lane width 



192			t_lane_width_f = vision_lane_info_st.hostLaneWidth;



193			t_lane_width_f = MAX_TP(t_lane_width_f, t_min_lane_width);



194			t_lane_width_f = MIN_TP(t_lane_width_f, t_max_lane_width);



195			m_best_line_info_st.m_lane_width[1] = t_lane_width_f;



196



197			if (m_debounce4MLane_f < 1)



198			{



199				m_debounce4MLane_f = 0;



200			}



201			else



202			{



203				m_debounce4MLane_f = m_debounce4MLane_f - veh_state.vRef * GetCycleTime_s();



204			}



205		}



206		else



207		{



208			if (m_best_line_info_st.m_lane_width[1] + FLOAT_EPSILON >= t_min_lane_width)// last cycle exist



209			{



210				if (m_debounce4MLane_f < MAX_DEBOUNCE_DIS_OF_VSL)



211				{



212					m_debounce4MLane_f += veh_state.vRef * GetCycleTime_s();



213				}



214				else



215				{



216					m_debounce4MLane_f = MAX_DEBOUNCE_DIS_OF_VSL;



217					m_best_line_info_st.m_lane_width[1] = 0;



218				}



219



220			}



221



222



223		}



224		//left lane width



225		if (((MEASUREMENT == vision_lane_info_st.lines_st_arr[0].traceStatus) || (VISION_LINE_MIDDLE_HIGH <= vision_lane_info_st.lines_st_arr[0].quality)) &&



226			((MEASUREMENT == vision_lane_info_st.lines_st_arr[1].traceStatus) || (VISION_LINE_MIDDLE_HIGH <= vision_lane_info_st.lines_st_arr[1].quality)))



227		{



228			t_lane_width_f = vision_lane_info_st.lines_st_arr[0].curFunction.latDistanceZeroOrderCoeff - vision_lane_info_st.lines_st_arr[1].curFunction.latDistanceZeroOrderCoeff;



229			if ((m_best_line_info_st.m_lane_width[0] >= t_min_lane_width) && //first time



230				(m_best_line_info_st.m_lane_width[0] <= t_max_lane_width))



231			{



232				t_lane_width_f = 0.5 * (m_best_line_info_st.m_lane_width[0] + t_lane_width_f);



233				t_lane_width_f = MAX_TP(t_lane_width_f, t_min_lane_width);



234				t_lane_width_f = MIN_TP(t_lane_width_f, t_max_lane_width);



235			}



236			m_best_line_info_st.m_lane_width[0] = t_lane_width_f;



237



238			if (m_debounce4LLane_f < 1)



239			{



240				m_debounce4LLane_f = 0;



241			}



242			else



243			{



244				m_debounce4LLane_f = m_debounce4LLane_f - veh_state.vRef * GetCycleTime_s();



245			}



246		}



247		else if ((MEASUREMENT == vision_lane_info_st.lines_st_arr[0].traceStatus) || (VISION_LINE_MIDDLE_HIGH <= vision_lane_info_st.lines_st_arr[0].quality))



248		{



249			t_lane_width_f = vision_lane_info_st.lines_st_arr[0].curFunction.latDistanceZeroOrderCoeff - 0.5 * m_best_line_info_st.m_lane_width[1];



250			if ((m_best_line_info_st.m_lane_width[0] >= t_min_lane_width - 0.5) && //first time



251				(m_best_line_info_st.m_lane_width[0] <= t_max_lane_width + 0.5))



252			{



253				t_lane_width_f = 0.5 * (m_best_line_info_st.m_lane_width[0] + t_lane_width_f);



254				t_lane_width_f = MAX_TP(t_lane_width_f, t_min_lane_width);



255				t_lane_width_f = MIN_TP(t_lane_width_f, t_max_lane_width);



256			}



257			m_best_line_info_st.m_lane_width[0] = t_lane_width_f;



258



259			if (m_debounce4LLane_f < 1)



260			{



261				m_debounce4LLane_f = 0;



262			}



263			else



264			{



265				m_debounce4LLane_f = m_debounce4LLane_f - veh_state.vRef * GetCycleTime_s();



266			}



267		}



268		else



269		{



270			if (m_best_line_info_st.m_lane_width[0] + FLOAT_EPSILON >= t_min_lane_width)// last cycle exist



271			{



272				if (m_debounce4LLane_f < MAX_DEBOUNCE_DIS_OF_VSL)



273				{



274					m_debounce4LLane_f += veh_state.vRef * GetCycleTime_s();



275				}



276				else



277				{



278					m_debounce4LLane_f = MAX_DEBOUNCE_DIS_OF_VSL;



279					m_best_line_info_st.m_lane_width[0] = 0;



280				}



281



282			}



283



284



285		}



286



287		//right lane width



288		if (((MEASUREMENT == vision_lane_info_st.lines_st_arr[2].traceStatus) || (VISION_LINE_MIDDLE_HIGH <= vision_lane_info_st.lines_st_arr[2].quality)) &&



289			((MEASUREMENT == vision_lane_info_st.lines_st_arr[3].traceStatus) || (VISION_LINE_MIDDLE_HIGH <= vision_lane_info_st.lines_st_arr[3].quality)))



290		{



291			t_lane_width_f = vision_lane_info_st.lines_st_arr[2].curFunction.latDistanceZeroOrderCoeff - vision_lane_info_st.lines_st_arr[3].curFunction.latDistanceZeroOrderCoeff;



292			if ((m_best_line_info_st.m_lane_width[2] >= t_min_lane_width) && //first time



293				(m_best_line_info_st.m_lane_width[2] <= t_max_lane_width))



294			{



295				t_lane_width_f = 0.5 * (m_best_line_info_st.m_lane_width[2] + t_lane_width_f);



296				t_lane_width_f = MAX_TP(t_lane_width_f, t_min_lane_width);



297				t_lane_width_f = MIN_TP(t_lane_width_f, t_max_lane_width);



298			}



299			m_best_line_info_st.m_lane_width[2] = t_lane_width_f;



300



301			if (m_debounce4RLane_f < 1)



302			{



303				m_debounce4RLane_f = 0;



304			}



305			else



306			{



307				m_debounce4RLane_f = m_debounce4RLane_f - veh_state.vRef * GetCycleTime_s();



308			}



309		}



310		else if ((MEASUREMENT == vision_lane_info_st.lines_st_arr[3].traceStatus) || (VISION_LINE_MIDDLE_HIGH <= vision_lane_info_st.lines_st_arr[3].quality))



311		{



312			t_lane_width_f = -vision_lane_info_st.lines_st_arr[3].curFunction.latDistanceZeroOrderCoeff - 0.5 * m_best_line_info_st.m_lane_width[1];



313			if ((m_best_line_info_st.m_lane_width[2] >= t_min_lane_width) && //first time



314				(m_best_line_info_st.m_lane_width[2] <= t_max_lane_width))



315			{



316				t_lane_width_f = 0.5 * (m_best_line_info_st.m_lane_width[2] + t_lane_width_f);



317				t_lane_width_f = MAX_TP(t_lane_width_f, t_min_lane_width);



318				t_lane_width_f = MIN_TP(t_lane_width_f, t_max_lane_width);



319			}



320			m_best_line_info_st.m_lane_width[2] = t_lane_width_f;



321



322			if (m_debounce4RLane_f < 1)



323			{



324				m_debounce4RLane_f = 0;



325			}



326			else



327			{



328				m_debounce4RLane_f = m_debounce4RLane_f - veh_state.vRef * GetCycleTime_s();



329			}



330		}



331		else



332		{



333



334			if (m_best_line_info_st.m_lane_width[2] + FLOAT_EPSILON >= t_min_lane_width)// last cycle exist



335			{



336				if (m_debounce4RLane_f < MAX_DEBOUNCE_DIS_OF_VSL)



337				{



338					m_debounce4RLane_f += veh_state.vRef * GetCycleTime_s();



339				}



340				else



341				{



342					m_debounce4RLane_f = MAX_DEBOUNCE_DIS_OF_VSL;



343					m_best_line_info_st.m_lane_width[2] = 0;



344				}



345



346			}



347		}



348



349	}

353void VSL::frontLaneModelCalc_v(LM_GENERAL_LIMIT_ST& inputLimits_pst, VISION_LANE_INFO_ST& vision_lane_info_st, VEHSTATE_ST& veh_state, VEHICLE_MOVEMENT_ST& t_vehMovement_st)



354{



355	uint8_t const t_bestLineNum = m_best_line_info_st.m_line_id_un8;



356	float t_lane_length_f = 0;



357



358	if (t_bestLineNum < VSL_MAX_LINE_NUM)// valid



359	{



360		float t_Lane_coeff_c0[VSL_MAX_LANE_NUM];



361		for (uint8_t t_i = 0; t_i < VSL_MAX_LANE_NUM; t_i++)



362		{



363			t_Lane_coeff_c0[t_i] = 0;



364



365		}



366		m_lanes_general_info_st.available_b = true;



367		// calc left line coeff c0



368		switch (m_best_line_info_st.m_line_id_un8)



369		{



370		case 0:



371			t_Lane_coeff_c0[0] = m_best_line_info_st.m_coeff_f[0];



372			t_Lane_coeff_c0[1] = m_best_line_info_st.m_coeff_f[0] - m_best_line_info_st.m_lane_width[0];



373			t_Lane_coeff_c0[2] = m_best_line_info_st.m_coeff_f[0] - m_best_line_info_st.m_lane_width[0] - m_best_line_info_st.m_lane_width[1];



374			break;



375		case 1:



376			t_Lane_coeff_c0[0] = m_best_line_info_st.m_coeff_f[0] + m_best_line_info_st.m_lane_width[0];



377			t_Lane_coeff_c0[1] = m_best_line_info_st.m_coeff_f[0];



378			t_Lane_coeff_c0[2] = m_best_line_info_st.m_coeff_f[0] - m_best_line_info_st.m_lane_width[1];



379			break;



380		case 2:



381			t_Lane_coeff_c0[0] = m_best_line_info_st.m_coeff_f[0] + m_best_line_info_st.m_lane_width[0] + m_best_line_info_st.m_lane_width[1];



382			t_Lane_coeff_c0[1] = m_best_line_info_st.m_coeff_f[0] + m_best_line_info_st.m_lane_width[1];



383			t_Lane_coeff_c0[2] = m_best_line_info_st.m_coeff_f[0];



384			break;



385		case 3:



386			t_Lane_coeff_c0[0] = m_best_line_info_st.m_coeff_f[0] + m_best_line_info_st.m_lane_width[0] + m_best_line_info_st.m_lane_width[1] + m_best_line_info_st.m_lane_width[2];



387			t_Lane_coeff_c0[1] = m_best_line_info_st.m_coeff_f[0] + m_best_line_info_st.m_lane_width[1] + m_best_line_info_st.m_lane_width[2];



388			t_Lane_coeff_c0[2] = m_best_line_info_st.m_coeff_f[0] + m_best_line_info_st.m_lane_width[2];
