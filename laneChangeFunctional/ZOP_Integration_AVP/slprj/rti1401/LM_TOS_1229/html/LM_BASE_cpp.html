23void LM_BASE::Init_FrontLane_v()



24{



25	m_lanes_general_info_st.available_b = false;



26	for (uint8_t t_lane = 0; t_lane < VSL_MAX_LANE_NUM; t_lane++)



27	{



28		m_lanes_general_info_st.lane_general_info_st[t_lane].available_b = false;



29		m_lanes_general_info_st.lane_general_info_st[t_lane].width_f = NULL_F;



30		for (uint8_t t_course = LM_START_NUMBER_OF_FRONT_COURSE_LIMITS; t_course < LM_MAX_NUMBER_OF_COURSE_LIMITS; t_course++)



31		{



32			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].available_b = false;



33			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].dlowerLimit_f = MAX_F;



34			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].dupperLimit_f = NULL_F;



35			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].detectedByVideo_b = false;



36			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].quality_f = NULL_F;



37			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].numOfSegment_un8 = MAX_UN8;



38			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].coef_arr_f[0] = MAX_F;



39			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].coef_arr_f[1] = MAX_F;



40			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].coef_arr_f[2] = MAX_F;



41			m_lanes_general_info_st.lane_general_info_st[t_lane].segments_st[t_course].coef_arr_f[3] = MAX_F;



42		}



43



44	}



45}
