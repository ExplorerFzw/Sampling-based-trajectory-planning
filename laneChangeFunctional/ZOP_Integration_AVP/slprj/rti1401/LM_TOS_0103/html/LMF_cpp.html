10void LMF::Init()



11{

18		lanemodel_output_st.lanes_st[t_lane].rear_lane_length_f = NULL_F;

24			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].dupperLimit_f = -MAX_F;

27			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].coef_arr_f[0] = NULL_F;

37void LMF::Calc_v(LANES_GENERAL_INFO_ST& vsl_lane_info_st)



38{



39	//directly from VSL



40	lanemodel_output_st.egoLaneID_un8 = MIDDLE_LANE;



41	for (uint8_t t_lane = LEFT_LANE, t_lane_vsl = 0; t_lane <= RIGHT_LANE; t_lane++, t_lane_vsl++)//currently just consider 3 lanes



42	{



43		lanemodel_output_st.lanes_st[t_lane].available_b = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].available_b;



44		lanemodel_output_st.lanes_st[t_lane].lane_dwidth_f = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].width_f;



45		lanemodel_output_st.lanes_st[t_lane].front_lane_length_f = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].front_length_f;



46		lanemodel_output_st.lanes_st[t_lane].rear_lane_length_f = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].rear_length_f;



47		for (uint8_t t_course = 0; t_course < LM_MAX_NUMBER_OF_COURSE_LIMITS; t_course++)



48		{



49			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].available_b = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].available_b;



50			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].detectedByVideo_b = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].detectedByVideo_b;



51			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].dlowerLimit_f = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].dlowerLimit_f;



52			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].dupperLimit_f = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].dupperLimit_f;



53			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].numOfSegment_un8 = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].numOfSegment_un8;



54			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].quality_f = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].quality_f;



55			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].coef_arr_f[0] = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].coef_arr_f[0];



56			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].coef_arr_f[1] = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].coef_arr_f[1];



57			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].coef_arr_f[2] = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].coef_arr_f[2];



58			lanemodel_output_st.lanes_st[t_lane].segments_st[t_course].coef_arr_f[3] = vsl_lane_info_st.lane_general_info_st[t_lane_vsl].segments_st[t_course].coef_arr_f[3];



59



60		}



61		



62	}
