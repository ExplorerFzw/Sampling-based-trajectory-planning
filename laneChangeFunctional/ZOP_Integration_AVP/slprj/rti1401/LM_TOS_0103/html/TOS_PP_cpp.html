161{

163	t_objWidth = MAX_TP(t_objWidth, t_minWidth);



164	



165	if ((t_d_r >= -t_laneHalfWidth) && (t_d_l <= t_laneHalfWidth)



166		)



167	{



168		t_probL_f = NULL_F;



169		t_probM_f = float(1.0);



170		t_probR_f = NULL_F;



171	}



172	else if (t_d_r < -t_laneHalfWidth)



173	{



174		t_probL_f = NULL_F;



175		t_probM_f = float((t_laneHalfWidth + t_d_l) / t_objWidth); 



176		t_probR_f = float(1.0 - t_probM_f);



177	}



178	else



179	{



180		t_probL_f = float((t_d_l - t_laneHalfWidth) / t_objWidth);



181		t_probM_f =  float(1.0 - t_probL_f);



182		t_probR_f = NULL_F;



183	}

472float TOS_PP::CutInProbCalc(uint8_t t_obj_num, FUSION_OBJ_OUTPUT_ST& fus_output_st, LANE_MODEL_OUTPUT_ST& laneModel_output_st)//just for object in R,L,M Lane

484		(t_obj_vy_f < -TOS_PP_CutInMinVy_f)



485		)

497		}

501		}



502	}

507		)

511		float t_dPre2CenterLane = t_dMovingAfterCertainTime + t_dis2EgoLaneCenter;



512		if (t_dPre2CenterLane >= -0.5 * t_obj_width_f)



513		{



514			t_prob_f = 1.0;



515		}

517		{



518			t_prob_f = 0.0;



519		}

674float TOS_PP::TTC_Calac(uint8_t t_index,FUSION_OBJ_OUTPUT_ST& fus_output_st, VEHSTATE_ST& t_vehState_st)



675{



676	float t_ttc = MAX_F;



677	float t_max = 60;



678	if ((true == m_tos_obj_within_lane_st.m_valid[t_index]) &&



679		(m_tos_obj_within_lane_st.m_withinLaneId[t_index] >= LEFT_LANE) &&

682		)



683	{

691		if (fabsf(t_dRel) <= t_dMin)



692		{



693			t_ttc = 0;



694		}



695		else if (t_dRel > t_dMin)// in front



696		{



697			if (t_vRel < -0.3)



698			{



699				t_ttc = -(t_dRel - t_dDrontBumper - 0.5 * t_dobjLength) / t_vRel;



700			}



701			else



702			{



703				t_ttc = MAX_F;



704			}



705		}



706		else//in rear



707		{



708			if (t_vRel > 0.3)



709			{



710				t_ttc = -(t_dRel + t_dDrontBumper + 0.5 * t_dobjLength) / t_vRel;



711			}



712			else



713			{



714				t_ttc = MAX_F;



715			}



716		}



717



718	}



719	else



720	{



721		t_ttc = MAX_F;



722	}



723



724	t_ttc = MIN_TP(t_ttc, t_max);



725	return t_ttc;



726



727}
