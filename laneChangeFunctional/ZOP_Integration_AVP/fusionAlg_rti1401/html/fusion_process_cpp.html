193/*******************************************************************************



194* @name        FusionAlg_Run_V



195* @brief



196* @param[in]   pHostMotion: the vehicle data (yawRate and speed) of the current time



197*			   pRadarInfo: the radar data



198*			   pVisObj: the output of OD



199*			   pVisLane: the output of LD



200* @param[out]  pFusOutInfo: the output of fusion



201* @return      ZONE_RETURN_OK: success



202*              ZONE_RETURN_ERR: fail



203* @see         none



204* @note        none



205******************************************************************************/



206



207int32_t FusionAlg_Run_V(const VehicleData      *pHostMotion,



208						const CanRadarObjList  *pRadarInfo,



209						const VisionObjList    *pVisObj,



210						const VisionLaneMark   *pVisLane,



211						const uint8_t		   runFlag,



212						FusionObjOutInfo       *pFusOutInfo)



213{



214	int32_t	status = ZONE_RETURN_OK;



215	uint8_t outputCnt = 0;



216



217	if (NULL == pHostMotion || NULL == pRadarInfo || NULL == pVisObj || NULL == pFusOutInfo)



218	{



219		ZONE_ERROR("Fusion", "Input Info Parameters Invalid!\r\n");



220		return ZONE_RETURN_ERR;



221	}



222



223	ZONE_INFO("FUSION_VER", "normal release: %d", 101);



224



225	memset(pFusOutInfo, 0, sizeof(FusionObjOutInfo));



226



227	if (runFlag == 0){



228		



229		status = SensorAdapter_Run_V(pHostMotion, pRadarInfo, pVisObj);



230



231		status = Sync_Run_V(pRadarInfo, pVisObj);



232



233		status = Association_Run_V();



234



235		status = ClusterManager_Run_V();



236



237		status = FusionTrackManager_Run_V();



238



239		status = OutPutAdapter_Run_V(pVisLane, pFusOutInfo);



240



241		gFrameNum++;



242



243		pFusOutInfo->timeStamp = pRadarInfo->timestamp;



244		pFusOutInfo->frameNum = gFrameNum;



245	}



246	else if (runFlag == 1){



247



248		gFrameNum++;



249



250		pFusOutInfo->timeStamp = pRadarInfo->timestamp;



251		pFusOutInfo->frameNum = gFrameNum;



252



253		for (uint8 radarCnt = 0; radarCnt < UTILS_RADAR_MAX_OBJ_NUM; radarCnt++) {



254			if (pRadarInfo->radarObj[radarCnt].objValidFlag == 1 && pRadarInfo->radarObj[radarCnt].objExstProb > 30 && outputCnt < FUSION_OBJ_OUT_MAX) {



255				pFusOutInfo->objects[outputCnt].obj_estimate.longPosition = pRadarInfo->radarObj[radarCnt].objXPos;



256				pFusOutInfo->objects[outputCnt].obj_estimate.latPosition = pRadarInfo->radarObj[radarCnt].objYPos;



257				pFusOutInfo->objects[outputCnt].obj_estimate.headingAngle = 0;



258				pFusOutInfo->objects[outputCnt].obj_estimate.speed = pRadarInfo->radarObj[radarCnt].objXVelRel;



259				pFusOutInfo->objects[outputCnt].obj_estimate.yawRate = 0;



260				pFusOutInfo->objects[outputCnt].obj_estimate.longRelativeVelocity = pRadarInfo->radarObj[radarCnt].objXVelRel;



261				pFusOutInfo->objects[outputCnt].obj_estimate.latRelativeVelocity = pRadarInfo->radarObj[radarCnt].objYVelRel;



262				pFusOutInfo->objects[outputCnt].obj_estimate.longAbsoluteVelocity = 0.0;



263				pFusOutInfo->objects[outputCnt].obj_estimate.latAbsoluteVelocity = 0.0;



264				pFusOutInfo->objects[outputCnt].obj_estimate.acceleration = pRadarInfo->radarObj[radarCnt].objXAccRel;



265				pFusOutInfo->objects[outputCnt].obj_estimate.longRelativeAcceleration = pRadarInfo->radarObj[radarCnt].objXAccRel;



266				pFusOutInfo->objects[outputCnt].obj_estimate.latRelativeAcceleration = 0;



267				pFusOutInfo->objects[outputCnt].obj_estimate.longAbsoluteAcceleration = 0;



268				pFusOutInfo->objects[outputCnt].obj_estimate.latAbsoluteAcceleration = 0;



269				pFusOutInfo->objects[outputCnt].obj_estimate.curvature = 0;



270



271				pFusOutInfo->objects[outputCnt].obj_information.valid = pRadarInfo->radarObj[radarCnt].objValidFlag;



272				if (pRadarInfo->radarObj[radarCnt].objType == 0x2 || pRadarInfo->radarObj[radarCnt].objType == 0x3)



273					pFusOutInfo->objects[outputCnt].obj_information.type = 1;



274				else if (pRadarInfo->radarObj[radarCnt].objType == 0x1)



275					pFusOutInfo->objects[outputCnt].obj_information.type = 3;



276				else



277					pFusOutInfo->objects[outputCnt].obj_information.type = 0;



278				pFusOutInfo->objects[outputCnt].obj_information.nearestSide = 0;



279				pFusOutInfo->objects[outputCnt].obj_information.motionPatternCurrent = pRadarInfo->radarObj[radarCnt].objMotionPattern;



280				pFusOutInfo->objects[outputCnt].obj_information.motionPatternHistory = 0;



281				pFusOutInfo->objects[outputCnt].obj_information.width = 0;



282				pFusOutInfo->objects[outputCnt].obj_information.height = 0;



283				pFusOutInfo->objects[outputCnt].obj_information.length = 0;



284				pFusOutInfo->objects[outputCnt].obj_information.distanceToleftNearLaneMarking = 0;



285				pFusOutInfo->objects[outputCnt].obj_information.distanceToRightNearLaneMarking = 0;



286



287				pFusOutInfo->objects[outputCnt].obj_property.id = pRadarInfo->radarObj[radarCnt].objID;



288				pFusOutInfo->objects[outputCnt].obj_property.fusionSource = 1;



289				pFusOutInfo->objects[outputCnt].obj_property.motionModel = 0;



290				pFusOutInfo->objects[outputCnt].obj_property.trackStatus = pRadarInfo->radarObj[radarCnt].objUpdateFlag;



291				pFusOutInfo->objects[outputCnt].obj_property.positionConfidence = pRadarInfo->radarObj[radarCnt].objExstProb * 0.01;



292				pFusOutInfo->objects[outputCnt].obj_property.fcwConfidence = 0;



293				pFusOutInfo->objects[outputCnt].obj_property.tjaConfidence = 0;



294				pFusOutInfo->objects[outputCnt].obj_property.leftNearLaneMarkingConfidence = 0;



295				pFusOutInfo->objects[outputCnt].obj_property.rightNearLaneMarkingConfidence = 0;



296				pFusOutInfo->objects[outputCnt].obj_property.visionId = 0;



297				pFusOutInfo->objects[outputCnt].obj_property.accelerationStdDev = 0;



298				pFusOutInfo->objects[outputCnt].obj_property.headingAngleStdDev = 0;



299				pFusOutInfo->objects[outputCnt].obj_property.latPositionStdDev = pRadarInfo->radarObj[radarCnt].objYPos_Stdev;



300				pFusOutInfo->objects[outputCnt].obj_property.longPositionStdDev = pRadarInfo->radarObj[radarCnt].objXPos_Stdev;



301				pFusOutInfo->objects[outputCnt].obj_property.speedStdDev = pRadarInfo->radarObj[radarCnt].objXVelRel_Stdev;



302



303				outputCnt++;



304			}



305		}



306	}



307	else {



308



309	}



310	



311	return 0;
