85/**



86*******************************************************************************



87* \brief



88*



89*     - compute the associate param between the radar obj and the vision obj



90*



91* \param  pRadarTrack     [IN]  radar internal info



92*			pVisionTrack     [IN]  vision internal info



93*



94* \return status   FUSION_STATUS_OK on success



95*



96*******************************************************************************



97*/



98float CalAssociatePro(const RadarCluster *pClusterDetail,



99	const VisTrackDetail* pVisionTrack, float32 &MinX)



100{



101	uint32 i_ClusterID = pClusterDetail->ID;



102	float32 d_pro_pre = pVisionTrack->probOfAssociation[i_ClusterID - 1];



103	float32 d_long_radar = pClusterDetail->f_x;



104	//calc the param X



105	float32 d_delta_plong = fabs(d_long_radar - pVisionTrack->obj_coordinate.longitudinal);//||X||



106	float32 d_delta_plong2 = fabs(d_long_radar - pVisionTrack->obj_coordinate.longitudinal - gFusionAlgParameter.ASSOCIATION_VISSYSXERR);//||X||



107	float32 d_threshold_long;



108	if (d_long_radar > LONG_BOUND1) {



109		d_threshold_long = d_long_radar*0.5F;



110	}



111	else if (pVisionTrack->obj_coordinate.longitudinal < LONG_BOUND2) {



112		d_threshold_long = LONG_THRESHOLD2;



113	}



114	else {



115		d_threshold_long = LONG_THRESHOLD1;



116	}



117	//calc the param angle



118	float32 visionTheta = atan2f(pVisionTrack->obj_coordinate.lateral, pVisionTrack->obj_coordinate.longitudinal);



119	float32 radarTheta = atan2f(pClusterDetail->f_y, d_long_radar);



120	float32 d_delta_angle = fabs(radarTheta - visionTheta);//|angle|;



121	if (pVisionTrack->obj_type == VISION_TYPE_TRUCK)



122	{



123		float32 yMid = d_long_radar*pVisionTrack->obj_coordinate.lateral / pVisionTrack->obj_coordinate.longitudinal;



124		float32 thetaMax_truck = atan2f(yMid + 1.0F, d_long_radar);



125		float32 thetaMin_truck = atan2f(yMid - 1.0F, d_long_radar);



126		if (radarTheta > thetaMax_truck || radarTheta < thetaMin_truck)



127		{



128		}



129		else



130		{



131			d_delta_angle = 0.7F*d_delta_angle;



132		}



133		thetaMax_truck = atan2f(yMid + 0.75F, d_long_radar);



134		thetaMin_truck = atan2f(yMid - 0.75F, d_long_radar);



135		if ((radarTheta < thetaMax_truck) && (radarTheta > thetaMin_truck) && (MinX > d_long_radar))



136		{



137			MinX = d_long_radar;



138		}



139	}



140	if ((d_delta_angle > ANGLE_THRESHOLD1 && d_long_radar > 10.0F) || (d_delta_angle > ANGLE_THRESHOLD2 && d_long_radar <= 10.0F))



141	{



142		return (d_pro_pre * gFusionAlgParameter.ASSOCIATION_KPRESCORE);



143	}



144	if (d_delta_plong > d_threshold_long)



145	{



146		return (d_pro_pre *  gFusionAlgParameter.ASSOCIATION_KPRESCORE);



147	}



148



149	//calc the param Y



150	float32 d_delta_plat = fabs(pClusterDetail->f_y - pVisionTrack->obj_coordinate.lateral);//||Y||



151	float32 d_threshold_lat = LAT_THRESHOLD_1 + d_long_radar * LAT_THRESHOLD_2;//0.017455 = tan(1)



152



153	if (d_delta_plat > d_threshold_lat)



154	{



155		return (d_pro_pre *  gFusionAlgParameter.ASSOCIATION_KPRESCORE);



156	}



157



158	//calc the param V



159	float32 d_delta_v = sqrtf((pClusterDetail->f_vx - pVisionTrack->obj_velocity.Vx) * (pClusterDetail->f_vx - pVisionTrack->obj_velocity.Vx)



160		+ (pClusterDetail->f_vy - pVisionTrack->obj_velocity.Vy) * (pClusterDetail->f_vy - pVisionTrack->obj_velocity.Vy));//||v||



161																														   //calc the param V



162	float32 d_delta_vx = fabs(pClusterDetail->f_vx - pVisionTrack->obj_velocity.Vx);



163	float32 d_delta_vy = fabs(pClusterDetail->f_vy - pVisionTrack->obj_velocity.Vy);



164	if (d_delta_vx > 20.0F || d_delta_vy > 5.0F)



165	{



166		return (d_pro_pre *  gFusionAlgParameter.ASSOCIATION_KPRESCORE);



167	}



168



169



170



171	float32  d_ttc = pVisionTrack->obj_TTC;



172	d_ttc = 0.0F;



173



174	uint8	 d_classification = (pClusterDetail->objType == pVisionTrack->obj_type) ? 1 : 0;



175	float32  d_obstacleProb;



176	if (pClusterDetail->objObstacleProb > 60.0F)



177	{



178		d_obstacleProb = 0.0F;



179	}



180	else if (pClusterDetail->objObstacleProb > 30.0F)



181	{



182		d_obstacleProb = 0.5F;



183	}



184	else if (pClusterDetail->objObstacleProb > 15.0F)



185	{



186		d_obstacleProb = 1.0F;



187	}



188	else



189	{



190		d_obstacleProb = 5.0F;



191	}



192



193	float32 d_cal = 0.0F;



194



195	//float32 k_angle3 = d_long_radar+50.0F;



196	if (d_long_radar < gFusionAlgParameter.ASSOCIATION_XSLOGTHRESCAR)



197	{



198		d_cal = gFusionAlgParameter.ASSOCIATION_KSXCAR * d_delta_plong +



199			gFusionAlgParameter.ASSOCIATION_KSYCAR * d_delta_plat +



200			gFusionAlgParameter.ASSOCIATION_KSSPEEDCAR *d_delta_v +



201			gFusionAlgParameter.ASSOCIATION_KSTTCCAR * d_ttc +



202			gFusionAlgParameter.ASSOCIATION_KSANGLECAR * d_delta_angle +



203			gFusionAlgParameter.ASSOCIATION_KOBSPROB* d_obstacleProb -



204			gFusionAlgParameter.ASSOCIATION_KCLASSIFICATION* d_classification -



205			gFusionAlgParameter.ASSOCIATION_KPRESCORE * d_pro_pre;



206	}



207	else if (d_long_radar < gFusionAlgParameter.ASSOCIATION_XMLOGTHRESCAR)



208	{



209		d_cal = gFusionAlgParameter.ASSOCIATION_KMXCAR* d_delta_plong +



210			gFusionAlgParameter.ASSOCIATION_KMYCAR* d_delta_plat +



211			gFusionAlgParameter.ASSOCIATION_KMSPEEDCAR * d_delta_v +



212			gFusionAlgParameter.ASSOCIATION_KMTTCCAR * d_ttc +



213			(gFusionAlgParameter.ASSOCIATION_KMANGLECAR + d_long_radar)* d_delta_angle +



214			gFusionAlgParameter.ASSOCIATION_KOBSPROB * d_obstacleProb -



215			gFusionAlgParameter.ASSOCIATION_KCLASSIFICATION * d_classification -



216			gFusionAlgParameter.ASSOCIATION_KPRESCORE * d_pro_pre;



217



218	}



219	else if (d_long_radar < gFusionAlgParameter.ASSOCIATION_XBLOGTHRESCAR)



220	{



221		d_cal = gFusionAlgParameter.ASSOCIATION_KBXCAR* d_delta_plong2 +



222			gFusionAlgParameter.ASSOCIATION_KBYCAR * d_delta_plat +



223			gFusionAlgParameter.ASSOCIATION_KBSPEEDCAR * d_delta_v +



224			gFusionAlgParameter.ASSOCIATION_KBTTCCAR * d_ttc +



225			(gFusionAlgParameter.ASSOCIATION_KBANGLECAR + d_long_radar)* d_delta_angle +



226			gFusionAlgParameter.ASSOCIATION_KOBSPROB * d_obstacleProb -



227			gFusionAlgParameter.ASSOCIATION_KCLASSIFICATION * d_classification -



228			gFusionAlgParameter.ASSOCIATION_KPRESCORE * d_pro_pre;



229



230	}



231	else



232	{



233		d_cal = gFusionAlgParameter.ASSOCIATION_KLXCAR * d_delta_plong2 +



234			gFusionAlgParameter.ASSOCIATION_KLYCAR * d_delta_plat +



235			gFusionAlgParameter.ASSOCIATION_KLSPEEDCAR * d_delta_v +



236			gFusionAlgParameter.ASSOCIATION_KLTTCCAR * d_ttc +



237			gFusionAlgParameter.ASSOCIATION_KLANGLECAR * d_delta_angle +



238			gFusionAlgParameter.ASSOCIATION_KOBSPROB * d_obstacleProb -



239			gFusionAlgParameter.ASSOCIATION_KCLASSIFICATION * d_classification -



240			gFusionAlgParameter.ASSOCIATION_KPRESCORE * d_pro_pre;



241	}



242



243	d_cal = (d_cal > 0.0F) ? d_cal : 0.0F;



244	float32 d_asspro = expf(-1.0F * d_cal / 10.0F);



245



246	return d_asspro;



247}



248/*******************************************************************************



249* @name        CalReferencePointState



250* @brief



251* @param[in]   none



252* @param[out]  none



253* @return      ZONE_RETURN_OK: success



254*              ZONE_RETURN_ERR: fail



255* @see         none



256* @note        none



257******************************************************************************/



258void CalReferencePointState() {



259	//cal the reference point of cluster



260	for (RadarClusterMap::iterator iterCluster = gFusionIntInfo.clusterDetail.begin(); iterCluster != gFusionIntInfo.clusterDetail.end(); ++iterCluster)



261	{



262		RadarCluster& refClusterDetail = iterCluster->second;



263		//assert(refClusterDetail.numOfRadar > 0);



264		refClusterDetail.isAssociate = false;



265		refClusterDetail.objType = VISION_TYPE_UNKNOWN;



266		refClusterDetail.objObstacleProb = 0.0F;



267		refClusterDetail.objExsitProb = 0.0F;



268



269		uint8 objTypeTemp = VISION_TYPE_UNKNOWN;



270		bool firstFlag = true;



271		for (radarIdMap::iterator iterRadarId = refClusterDetail.radarID.begin(); iterRadarId != refClusterDetail.radarID.end(); ++iterRadarId)



272		{



273			//RadarTrackDetail& refRadarTrackDetail = pFusionSystemInternalInfo->radarIntDetail.at(iterRadarId->first);



274			RadarTrackDetailMap::iterator iterRadar = gFusionIntInfo.radarIntDetail.find(iterRadarId->first);



275			//assert(gFusionIntInfo.radarIntDetail.end() != iterRadar);



276			RadarTrackDetail &refRadarTrackDetail = iterRadar->second;



277



278			if (refRadarTrackDetail.objType == 0) //unknown



279			{



280



281				objTypeTemp = VISION_TYPE_UNKNOWN;



282			}



283			else if (refRadarTrackDetail.objType == 1 || refRadarTrackDetail.objType == 2) //4wheels or 2wheels



284			{



285				objTypeTemp = VISION_TYPE_CAR;



286			}



287			else //pedestrian



288			{



289				objTypeTemp = VISION_TYPE_PEDESTRAIN;



290			}



291			if (objTypeTemp > refClusterDetail.objType)



292			{



293				refClusterDetail.objType = objTypeTemp;



294			}



295



296			if (refRadarTrackDetail.objObstacleProb > refClusterDetail.objObstacleProb)



297			{



298				refClusterDetail.objObstacleProb = refRadarTrackDetail.objObstacleProb;



299			}



300			if (refRadarTrackDetail.objExstProb > refClusterDetail.objExsitProb)



301			{



302				refClusterDetail.objExsitProb = refRadarTrackDetail.objExstProb;



303			}



304



305



306			//update x,y,vx,vy



307			if (firstFlag)



308			{



309				refClusterDetail.f_x = refRadarTrackDetail.objXPos;



310				refClusterDetail.f_y = refRadarTrackDetail.objYPos;



311				refClusterDetail.f_vx = refRadarTrackDetail.objXVelAbs;



312				refClusterDetail.f_vy = refRadarTrackDetail.objYVelAbs;



313				firstFlag = false;



314			}



315			else



316			{



317				refClusterDetail.f_x = ZONE_MIN(refClusterDetail.f_x, refRadarTrackDetail.objXPos);



318				refClusterDetail.f_y = refClusterDetail.f_y + refRadarTrackDetail.objYPos;



319				refClusterDetail.f_vx = refClusterDetail.f_vx + refRadarTrackDetail.objXVelAbs;



320				refClusterDetail.f_vy = refClusterDetail.f_vy + refRadarTrackDetail.objYVelAbs;



321			}



322		}



323		refClusterDetail.f_y = refClusterDetail.f_y / refClusterDetail.numOfRadar;



324		refClusterDetail.f_vx = refClusterDetail.f_vx / refClusterDetail.numOfRadar;



325		refClusterDetail.f_vy = refClusterDetail.f_vy / refClusterDetail.numOfRadar;



326	}



327}



328/*******************************************************************************



329* @name        void VisionClusterAssociation() {



330



331* @brief



332* @param[in]   none



333* @param[out]  none



334* @return      ZONE_RETURN_OK: success



335*              ZONE_RETURN_ERR: fail



336* @see         none



337* @note        none



338******************************************************************************/



339void VisionClusterAssociation() {



340	float32 f_pro = 0.0F;



341	float32 f_maxPro = 0.0F;



342	float32 longitudinal = 0.0F;



343	float32 lateral = 0.0F;



344



345	//[Louis] Put these thresholds as CALs, like K_ASSOCIATION_THRESHOLD



346	RadarCluster *targetClusterPtr = NULL;



347	RadarCluster *targetClusterPtr_preAsso = NULL;



348	float32 f_pro_preAsso = 0.0F;



349	for (VisTrackDetailMap::iterator iterVision = gFusionIntInfo.visIntDetail.begin(); iterVision != gFusionIntInfo.visIntDetail.end(); ++iterVision)



350	{



351		VisTrackDetail& refVisTrackDetail = iterVision->second;



352		f_maxPro = 0.0F;



353		f_pro_preAsso = 0.0F;



354		uint32	ClusterIDAsso = 0, ClusterIDAssoPre = 0;



355		float32 minX = 1000.0F;



356		uint8 noAssoFlag = 0;



357



358		if (refVisTrackDetail.obj_coordinate.longitudinalVariance >9.9F && (refVisTrackDetail.obj_type == VISION_TYPE_CAR || refVisTrackDetail.obj_type == VISION_TYPE_TRUCK))



359		{



360			noAssoFlag = 1;



361		}



362		if (!noAssoFlag)



363		{



364			//asso vision1 and cluster



365			if (refVisTrackDetail.obj_type == VISION_TYPE_TRUCK)



366			{



367				for (RadarClusterMap::iterator iterCluster = gFusionIntInfo.clusterDetail.begin(); iterCluster != gFusionIntInfo.clusterDetail.end(); ++iterCluster)



368				{



369					RadarCluster& refClusterDetail = iterCluster->second;



370



371					f_pro = CalAssociatePro(&refClusterDetail, &refVisTrackDetail, minX);



372



373					if ((gFusionIntInfo.dayNightState == VISION_NIGHT) && (refClusterDetail.objType == VISION_TYPE_UNKNOWN))



374					{



375						f_pro = f_pro * 0.9F;



376					}



377



378					refVisTrackDetail.probOfAssociation[refClusterDetail.ID - 1] = f_pro;



379					if (!refClusterDetail.isAssociate)



380					{



381						if (f_maxPro < f_pro)



382						{



383							if (refClusterDetail.f_x < (minX + 0.01F))



384							{



385								/*record the max score vision message*/



386								f_maxPro = f_pro;



387								targetClusterPtr = &refClusterDetail;



388								ClusterIDAsso = refClusterDetail.ID;



389							}



390						}



391						else if ((gFusionAlgParameter.ASSOCIATION_SCORETHRES < f_pro) && ((1000.0F - 0.01F) > minX) && (refClusterDetail.f_x < (minX + 0.01F)))



392						{



393							/*record the max score vision message*/



394							f_maxPro = f_pro;



395							targetClusterPtr = &refClusterDetail;



396							ClusterIDAsso = refClusterDetail.ID;



397						}



398					}



399



400					if ((refClusterDetail.isInvision == 1) && (refClusterDetail.visionID == refVisTrackDetail.obj_ID))



401					{



402						f_pro_preAsso = f_pro;



403						targetClusterPtr_preAsso = &refClusterDetail;



404						ClusterIDAssoPre = refClusterDetail.ID;



405					}



406				}



407				if ((f_maxPro > gFusionAlgParameter.ASSOCIATION_SCORETHRES) && (targetClusterPtr->f_x > (minX + 0.01F)))



408				{



409					f_maxPro = 0.7F;



410				}



411			}



412			else



413			{



414				for (RadarClusterMap::iterator iterCluster = gFusionIntInfo.clusterDetail.begin(); iterCluster != gFusionIntInfo.clusterDetail.end(); ++iterCluster)



415				{



416					RadarCluster& refClusterDetail = iterCluster->second;



417



418					f_pro = CalAssociatePro(&refClusterDetail, &refVisTrackDetail, minX);



419					if ((gFusionIntInfo.dayNightState == VISION_NIGHT) && (refClusterDetail.objType == VISION_TYPE_UNKNOWN))



420					{



421						f_pro = f_pro * 0.9F;



422					}



423					refVisTrackDetail.probOfAssociation[refClusterDetail.ID - 1] = f_pro;



424					if ((f_maxPro < f_pro) && (!refClusterDetail.isAssociate)) // is the score is recently max?



425					{



426						/*record the max score vision message*/



427						f_maxPro = f_pro;



428						targetClusterPtr = &refClusterDetail;



429						ClusterIDAsso = refClusterDetail.ID;



430					}



431



432					if ((refClusterDetail.isInvision == 1) && (refClusterDetail.visionID == refVisTrackDetail.obj_ID))



433					{



434						f_pro_preAsso = f_pro;



435						targetClusterPtr_preAsso = &refClusterDetail;



436						ClusterIDAssoPre = refClusterDetail.ID;



437					}



438				}



439			}



440		}



441



442		if (refVisTrackDetail.obj_type != VISION_TYPE_TRUCK)



443		{



444			if ((ClusterIDAssoPre != ClusterIDAsso) && (f_pro_preAsso > gFusionAlgParameter.ASSOCIATION_SCORETHRES) && (f_maxPro - f_pro_preAsso < 0.12F))



445			{



446				f_maxPro = f_pro_preAsso;



447				targetClusterPtr = targetClusterPtr_preAsso;



448			}



449		}



450



451



452		if ((f_maxPro >  gFusionAlgParameter.ASSOCIATION_SCORETHRES) && (targetClusterPtr->visionID == refVisTrackDetail.obj_ID)



453			&& (targetClusterPtr->isInvision == 1))



454		{



455			/*if the previous associated vision track can associate this radar track again, keep the association*/



456			targetClusterPtr->isAssociate = true;



457		}



458		else if (f_maxPro >  gFusionAlgParameter.ASSOCIATION_SCORETHRES)



459		{



460			/*is this radar track associate another vision track, delete the previous association and establish a new one*/



461			targetClusterPtr->isAssociate = true;



462			if ((0 != ClusterIDAssoPre) && (targetClusterPtr_preAsso != NULL) && (targetClusterPtr_preAsso->isInFusion))



463			{



464				uint32 fusionID = targetClusterPtr_preAsso->fusionID; // get associated fusion id																			



465				FusionObjInternalStateMap::iterator iterFus = gFusionIntInfo.fusObjIntState.find(fusionID);



466				//assert(gFusionIntInfo.fusObjIntState.end() != iterFus);



467				iterFus->second.visionConfirmCnt = 0;



468			}



469



470			if (refVisTrackDetail.isClusterInside)



471			{



472				delVisionInCluster(refVisTrackDetail, gFusionIntInfo);



473				delVisionInFusion(refVisTrackDetail, gFusionIntInfo);



474			}



475



476			if (targetClusterPtr->isInFusion)



477			{



478				if (targetClusterPtr->isInvision)



479				{



480					//VisTrackDetail& refVis = pFusionSystemInternalInfo->visIntDetail.at(targetClusterPtr->visionID);



481					VisTrackDetailMap::iterator visIter = gFusionIntInfo.visIntDetail.find(targetClusterPtr->visionID);



482					//assert(gFusionIntInfo.visIntDetail.end() != visIter);



483					VisTrackDetail& refVis = visIter->second;



484



485					delVisionInCluster(refVis, gFusionIntInfo);



486					delVisionInFusion(refVis, gFusionIntInfo);



487				}



488				if (refVisTrackDetail.isInFusionObj)



489				{



490					uint32 fusionID = refVisTrackDetail.associatedFusObjID; // get associated vision id																			



491					FusionObjInternalStateMap::iterator iterFus = gFusionIntInfo.fusObjIntState.find(fusionID);



492					//assert(gFusionIntInfo.fusObjIntState.end() != iterFus);



493



494					delVisionInFusion(refVisTrackDetail, gFusionIntInfo);



495



496					Del_Empty_Track(&gFusionIntInfo, iterFus);



497				}



498



499



500				refVisTrackDetail.isClusterInside = 1;



501				refVisTrackDetail.isInFusionObj = 1;



502				refVisTrackDetail.associatedClusterID = targetClusterPtr->ID;



503				refVisTrackDetail.associatedFusObjID = targetClusterPtr->fusionID;



504



505				//FusionObjInternalState& refFusion = pFusionSystemInternalInfo->fusObjIntState.at(targetClusterPtr->fusionID);



506				FusionObjInternalStateMap::iterator fusIter = gFusionIntInfo.fusObjIntState.find(targetClusterPtr->fusionID);



507				//assert(gFusionIntInfo.fusObjIntState.end() != fusIter);



508				FusionObjInternalState& refFusion = fusIter->second;



509



510				refFusion.isVisInside = 1;



511				refFusion.visID = refVisTrackDetail.obj_ID;



512				targetClusterPtr->isInvision = 1;



513				targetClusterPtr->visionID = refVisTrackDetail.obj_ID;



514			}



515			else



516			{



517				delClusterInVision(targetClusterPtr, gFusionIntInfo);



518



519				if (refVisTrackDetail.isInFusionObj)



520				{



521					//FusionObjInternalState& refFusion = pFusionSystemInternalInfo->fusObjIntState.at(refVisTrackDetail.associatedFusObjID);



522					FusionObjInternalStateMap::iterator fusIter = gFusionIntInfo.fusObjIntState.find(refVisTrackDetail.associatedFusObjID);



523					//assert(gFusionIntInfo.fusObjIntState.end() != fusIter);



524					FusionObjInternalState& refFusion = fusIter->second;



525



526					refFusion.isClusterInside = 1;



527					refFusion.clusterID = targetClusterPtr->ID;



528					refVisTrackDetail.isClusterInside = 1;



529					refVisTrackDetail.associatedClusterID = targetClusterPtr->ID;



530



531					targetClusterPtr->isInvision = 1;



532					targetClusterPtr->isInFusion = 1;



533					targetClusterPtr->visionID = refVisTrackDetail.obj_ID;



534					targetClusterPtr->fusionID = refFusion.id;



535				}
