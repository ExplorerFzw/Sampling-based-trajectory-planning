69/*******************************************************************************



70* @name        OutPutAdapter_Run_V



71* @brief



72* @param[in]   pVisionLane : vision lane info



73* @param[out]  pFusionObjOutInfo : fusion output info



74* @return      ZONE_RETURN_OK: success



75*              ZONE_RETURN_ERR: fail



76* @see         none



77* @note        none



78******************************************************************************/



79



80int32_t OutPutAdapter_Run_V(const VisionLaneMark* pVisionLane, FusionObjOutInfo* pFusionObjOutInfo)



81{



82	float32						cnow			= 0.0F;



83	LaneParam					laneParameter	= {0.0F};



84	FusionSystemInternalInfo*	pFusIntInfo		= &gFusionIntInfo;



85



86	// Compute the priority of each track



87	OutputAdapter_FusionTrackPriority();



88



89	// Sort the object by its priority



90	// This is a vector of {priority_score, object_ID} pairs



91	std::vector<std::pair<float32, uint32> > priority_id_pair;



92	priority_id_pair.reserve(pFusIntInfo->numOfActiveObj);



93	for (FusionObjInternalStateMap::iterator iterFusion = pFusIntInfo->fusObjIntState.begin(); iterFusion != pFusIntInfo->fusObjIntState.end(); ++iterFusion) {



94		priority_id_pair.push_back(std::make_pair(iterFusion->second.priority_score, iterFusion->first));



95	}



96	// Sorting will put lower priority_score ahead of larger ones (pair->first)



97	// The index is stored in pair->second.



98	sort(priority_id_pair.begin(), priority_id_pair.end());



99



100	laneParameter.Lvalid	=	pVisionLane->leftHostIndividual.lane_traceStatus;



101	laneParameter.LC0		=	pVisionLane->leftHostIndividual.lane_curve.latDistanceZeroOrderCoeff;



102	laneParameter.LC1		=	pVisionLane->leftHostIndividual.lane_curve.latDistanceFirstOrderCoeff;



103	laneParameter.LC2		=	pVisionLane->leftHostIndividual.lane_curve.latDistanceSecondOrderCoeff;



104	laneParameter.LC3		=	pVisionLane->leftHostIndividual.lane_curve.latDistanceThirdOrderCoeff;



105	laneParameter.LL		=	pVisionLane->leftHostIndividual.lane_curve.longDistanceToend;



106	laneParameter.Rvalid	=	pVisionLane->rightHostIndividual.lane_traceStatus;



107	laneParameter.RC0		=	pVisionLane->rightHostIndividual.lane_curve.latDistanceZeroOrderCoeff;



108	laneParameter.RC1		=	pVisionLane->rightHostIndividual.lane_curve.latDistanceFirstOrderCoeff;



109	laneParameter.RC2		=	pVisionLane->rightHostIndividual.lane_curve.latDistanceSecondOrderCoeff;



110	laneParameter.RC3		=	pVisionLane->rightHostIndividual.lane_curve.latDistanceThirdOrderCoeff;



111	laneParameter.RL		=	pVisionLane->rightHostIndividual.lane_curve.longDistanceToend;



112



113	if (fabs(pFusIntInfo->egoSpeedAndYawrate.EgoSpeedVal) < THRESHOS_EGO_SPEED)



114	{



115		cnow = 0.0F;



116	}



117	else



118	{



119		cnow = pFusIntInfo->egoSpeedAndYawrate.EgoYawRateVal / pFusIntInfo->egoSpeedAndYawrate.EgoSpeedVal;



120	}



121



122	// Copy the fusion internal state to the output structure



123	const uint8 outPutCnt = (pFusIntInfo->numOfActiveObj < FUSION_OBJ_OUT_MAX) ? pFusIntInfo->numOfActiveObj : FUSION_OBJ_OUT_MAX;



124	uint8 recentOutCnt = 0;



125	for (uint8 i = 0; i < pFusIntInfo->numOfActiveObj; i++) 



126	{



127		if (outPutCnt <= recentOutCnt)



128		{



129			break;



130		}



131		// Get the object ID from highest priority



132		uint32 id = priority_id_pair[i].second;



133



134		FusionObjInternalStateMap::iterator iterFus = pFusIntInfo->fusObjIntState.find(id);



135		ZONE_Assert(pFusIntInfo->fusObjIntState.end() != iterFus);



136		FusionObjInternalState& refFusionInternal = iterFus->second;



137		float32 dltPosY		= refFusionInternal.predLatPosn - refFusionInternal.latPosn;



138		float32 dltPosX		= refFusionInternal.predLongPosn - refFusionInternal.longPosn;



139		float32 dltVx		= refFusionInternal.predVx - refFusionInternal.longVel;



140		float32 dltVy		= refFusionInternal.predVy - refFusionInternal.latVel;



141		float32 dltSpeed	= refFusionInternal.predSpeed - refFusionInternal.speed;



142		float32 dltAcc		= refFusionInternal.predAcceleration - refFusionInternal.acceleration;



143		if (TRACK_STATUS_NEW != refFusionInternal.trackStatus)



144		{



145			if ((-9.0F > dltPosY		|| 9.0F < dltPosY)



146				|| (-9.0F > dltPosX		|| 9.0F < dltPosX) 



147				|| (-9.0F > dltVx		|| 9.0F < dltVx)



148				|| (-9.0F > dltVy		|| 9.0F < dltVy)



149				|| (-9.0F > dltSpeed	|| 9.0F < dltSpeed)



150				|| (-14.0F > dltAcc		|| 14.0F < dltAcc))



151			{



152				if (0 < refFusionInternal.oID)



153				{



154					pFusIntInfo->unassignedOutputID[pFusIntInfo->numOfUnassignedOutputID] = refFusionInternal.oID;



155					pFusIntInfo->numOfUnassignedOutputID++;



156					refFusionInternal.oID = 0;



157				}



158				continue;



159			}



160		}



161		uint8 inROIflag_Lane = 0, inROIflag_Ego = 0;



162		inROIflag_Lane = OutputAdapter_ROIdefineLane(refFusionInternal.longPosn, refFusionInternal.latPosn, cnow, laneParameter);



163		inROIflag_Ego = OutputAdapter_ROIdefineEgo(refFusionInternal.longPosn, refFusionInternal.latPosn, cnow);



164		if (inROIflag_Ego == 0 && inROIflag_Lane == 0)



165		{



166			if (0 < refFusionInternal.oID && (!refFusionInternal.f_is_moveable))



167			{



168				pFusIntInfo->unassignedOutputID[pFusIntInfo->numOfUnassignedOutputID] = refFusionInternal.oID;



169				pFusIntInfo->numOfUnassignedOutputID++;



170				refFusionInternal.oID = 0;



171			}



172			continue;



173		}



174



175		recentOutCnt++;



176



177		if (0 < refFusionInternal.oID) {



178			// If this object is already selected previously for output (i.e., the oID is >0)



179			// Later copy this object to the corresponding location of the output array.



180			refFusionInternal.opTrackStatus = refFusionInternal.trackStatus;



181		}



182		else {



183			// If this object is not output previously (i.e., does not have an oID yet).



184			if (0 < pFusIntInfo->numOfUnassignedOutputID) {



185				// If there is still ID available for output



186				// Get the available oID from the last position of the array



187				uint8 oID = pFusIntInfo->unassignedOutputID[pFusIntInfo->numOfUnassignedOutputID - 1];



188				// Reset this oID.



189				pFusIntInfo->unassignedOutputID[pFusIntInfo->numOfUnassignedOutputID - 1] = 0;



190				// Reduce the number of available oID



191				pFusIntInfo->numOfUnassignedOutputID--;



192				// Assign this oID to this fusion object.



193				refFusionInternal.oID = oID;



194			}



195			else



196			{



197				// 20171128==>TODO: Need to improve this by kicking out a lower priority track if there is a



198				// higher priority track coming in but no output slot available.



199				for (std::vector<std::pair<float32, uint32> >::reverse_iterator iterPriority = priority_id_pair.rbegin(); iterPriority != priority_id_pair.rend(); ++iterPriority)



200				{



201					uint32 id = iterPriority->second;



202



203					FusionObjInternalStateMap::iterator iterFus = pFusIntInfo->fusObjIntState.find(id);



204					ZONE_Assert(pFusIntInfo->fusObjIntState.end() != iterFus);



205					FusionObjInternalState& refLowPriorityFusion = iterFus->second;



206					if (0 < refLowPriorityFusion.oID)



207					{



208						refFusionInternal.oID = refLowPriorityFusion.oID;



209						refLowPriorityFusion.oID = 0;



210						break;



211					}



212				}



213			}



214



215			// 20171108==>TODO: Need to add an output_object_status. Usually this status is



216			// the same as the object status, however, we need to set output_object_status as NEW



217			// regardless of the object status when this is the first cycle of reporting this object



218			// to FUNCTION.



219			refFusionInternal.opTrackStatus = TRACK_STATUS_NEW; // NEW



220		}



221	}



222



223	OutputAdapter_CopyFusionInternalStateToOutput(pFusionObjOutInfo);



224	return ZONE_RETURN_OK;



225}

227/*******************************************************************************



228* @name        OutputAdapter_FusionTrackPriority



229* @brief       Compute the fusion track's priority



230*			   Prioritize the fusion track based on following:



231*			   1. Distance, closer target should have higher priority



232*			   2. Motion status, moving target should have higher priority



233*			   3. Track quality/confidence, high confidence target should have higher priority



234*			   4. UPDATED track should have higher priority, NEW/COASTED should have lower priority



235*			   5. Radar-Vision object should have higher priority. Radar or vision only object should have lower priority



236*			   6. Others



237* @param[in]   none



238* @param[out]  none



239* @return      ZONE_RETURN_OK: success



240*              ZONE_RETURN_ERR: fail



241* @see         none



242* @note        none



243******************************************************************************/



244



245int8_t OutputAdapter_FusionTrackPriority()



246{



247	const float32				K_STATIC_TRACK_PENALTY		= 50.0F;



248	const float32				K_STOPPED_TRACK_PENALTY		= 20.0F;



249	const float32				K_MED_QUALITY_TRACK_PENALTY	= 50.0F;



250	const float32				K_LOW_QUALITY_TRACK_PENALTY	= 100.0F;



251	const float32				K_MAX_PENALTY				= 1000000000.0F;



252	float32						priority_score				= 0.0F;



253	FusionSystemInternalInfo*	pFusIntInfo					= &gFusionIntInfo;



254



255	



256	for (FusionObjInternalStateMap::iterator iterFusion = pFusIntInfo->fusObjIntState.begin(); iterFusion != pFusIntInfo->fusObjIntState.end(); ++iterFusion) {



257		// Loop over the object array.



258		FusionObjInternalState& fus_obj = iterFusion->second;



259		priority_score = 0.0F;



260		if (TRACK_STATUS_INVALID != fus_obj.trackStatus) {



261			// For valid object



262			// The lower the score, the higher the priority.



263			// Add penalty for longitudinal distance.



264			priority_score += fus_obj.longPosn;



265			// 20171106==>TODO: Add penalties for other factors.



266		}



267		else {



268			// For invalid object.



269			priority_score = K_MAX_PENALTY;



270		}



271		// Assign the priority score to the object



272		fus_obj.priority_score = priority_score;



273	}



274	return ZONE_RETURN_OK;



275}

277/*******************************************************************************



278* @name        OutputAdapter_ROIdefineEgo



279* @brief



280* @param[in]   none



281* @param[out]  none



282* @return      ZONE_RETURN_OK: success



283*              ZONE_RETURN_ERR: fail



284* @see         none



285* @note        none



286******************************************************************************/



287



288uint8 OutputAdapter_ROIdefineLane(float32 objPosX, float32 objPosY, float32 Cnow, LaneParam& laneParameter)



289{



290	uint8 inROIflag = 0;



291



292	const  float32 Cappend  = 0.0005F;



293	const  float32 LaneW	= 3.5F;



294	const  float32 deltaC	= 0.0013F;



295	static float32 CTarget	= 0.0F;



296	float32_t dC		= 0.0F;



297	float32_t yl		= 0.0F;



298	float32_t yr		= 0.0F;



299	float32_t LC0, LC1, LC2, LC3, RC0, RC1, RC2, RC3;



300	float32_t objPosX2 = objPosX*objPosX;



301	float32_t objPosX3 = objPosX*objPosX2;



302



303	//Left Lane valid, right lane valid



304	if (laneParameter.Lvalid != 0 && laneParameter.Rvalid != 0)



305	{



306		LC0 = laneParameter.LC0;



307		LC1 = laneParameter.LC1;



308		LC2 = laneParameter.LC2;



309		LC3 = laneParameter.LC3;



310		RC0 = laneParameter.RC0;



311		RC1 = laneParameter.RC1;



312		RC2 = laneParameter.RC2;



313		RC3 = laneParameter.RC3;



314		if (objPosX < laneParameter.LL && objPosX < laneParameter.RL)



315		{



316			yl = LC0 + LC1 * objPosX + LC2 * objPosX2 + LC3 * objPosX3 + LaneW;



317			yr = RC0 + RC1 * objPosX + RC2 * objPosX2 + RC3 * objPosX3 - LaneW;



318		}



319		else if (objPosX < laneParameter.LL)



320		{



321			yl = LC0 + LC1 * objPosX + LC2 * objPosX2 + LC3 * objPosX3 + LaneW;



322			yr = yl - 3.0F * LaneW;



323		}



324		else if (objPosX < laneParameter.RL)



325		{



326			yr = RC0 + RC1 * objPosX + RC2 * objPosX2 + RC3 * objPosX3 - LaneW;



327			yl = yr + 3.0F * LaneW;			



328		}



329		else



330		{



331			// make sure two lanes do not cross 



332			if (LC2 < RC2)



333			{



334				RC2 = LC2;



335			}



336			if (LC3 < RC3)



337			{



338				if (LC3 > 0)



339				{



340					RC3 = LC3;



341				}



342				else



343				{



344					LC3 = RC3;



345				}



346			}



347			yl = LC0 + LC1 * objPosX + (LC2 + Cappend) * objPosX2 + LC3 * objPosX3 + LaneW;



348			yr = RC0 + RC1 * objPosX + (RC2 - Cappend) * objPosX2 + RC3 * objPosX3 - LaneW;



349		}



350



351		CTarget = ZONE_MIN(2.0F*LC2, 2.0F*RC2);



352	}



353	else if (laneParameter.Lvalid != 0) //Left Lane valid



354	{



355		LC0 = laneParameter.LC0;



356		LC1 = laneParameter.LC1;



357		LC2 = laneParameter.LC2;



358		LC3 = laneParameter.LC3;



359		if (objPosX < laneParameter.LL)



360		{



361			yl = LC0 + LC1 * objPosX + LC2 * objPosX2 + LC3 * objPosX3 + LaneW;



362			yr = LC0 + LC1 * objPosX + (LC2 - Cappend) * objPosX2 + LC3 * objPosX3 - 2 * LaneW;



363		}



364		else



365		{



366			yl = LC0 + LC1 * objPosX + (LC2 + Cappend) * objPosX2 + LC3 * objPosX3 + LaneW;



367			yr = LC0 + LC1 * objPosX + (LC2 - Cappend) * objPosX2 + LC3 * objPosX3 - 2 * LaneW;



368		}



369



370		CTarget = 2.0F * LC2;



371	}



372	else if (laneParameter.Rvalid != 0) //right Lane valid



373	{



374		RC0 = laneParameter.RC0;



375		RC1 = laneParameter.RC1;



376		RC2 = laneParameter.RC2;



377		RC3 = laneParameter.RC3;



378		if (objPosX < laneParameter.RL)



379		{



380			yl = RC0 + RC1 * objPosX + (RC2 + Cappend) * objPosX2 + RC3 * objPosX3 + 2 * LaneW;



381			yr = RC0 + RC1 * objPosX + RC2 * objPosX2 + RC3 * objPosX3 - LaneW;



382		}



383		else



384		{



385			yl = RC0 + RC1 * objPosX + (RC2 + Cappend) * objPosX2 + RC3 * objPosX3 + 2 * LaneW;



386			yr = RC0 + RC1 * objPosX + (RC2 - Cappend) * objPosX2 + RC3 * objPosX3 - LaneW;



387		}



388



389		CTarget = 2.0F * RC2;



390	}



391	else



392	{		



393		dC = Cnow - CTarget;



394		if (dC > deltaC)



395		{



396			CTarget += deltaC;



397		}



398		else if (dC < -deltaC)



399		{



400			CTarget -= deltaC;



401		}



402



403		yr = (0.5F * CTarget - Cappend)*objPosX2 - 1.5F * LaneW;



404		yl = (0.5F * CTarget + Cappend)*objPosX2 + 1.5F * LaneW;



405	}



406



407	



408	if (objPosY > yl || objPosY < yr)



409	{



410		inROIflag = 0;



411	}



412	else



413	{



414		inROIflag = 1;



415	}



416	return inROIflag;



417}

419/*******************************************************************************



420* @name        OutputAdapter_ROIdefineEgo



421* @brief



422* @param[in]   none



423* @param[out]  none



424* @return      ZONE_RETURN_OK: success



425*              ZONE_RETURN_ERR: fail



426* @see         none



427* @note        none



428******************************************************************************/



429



430uint8 OutputAdapter_ROIdefineEgo(float32 objPosX, float32 objPosY, float32 Cnow)



431{



432	uint8 inROIflag = 0;



433



434	const  float32 Cappend = 0.0005F;



435	const  float32 LaneW = 3.5F;



436	const  float32 deltaC = 0.0013F;



437	static float32 CTarget = 0.0F;



438	float32 dC = 0.0F;



439	float32 yl = 0.0F;



440	float32 yr = 0.0F;



441	float32 objPosX2 = objPosX*objPosX;



442	



443	dC = Cnow - CTarget;



444	if (dC > deltaC)



445	{



446		CTarget += deltaC;



447	}



448	else if (dC < -deltaC)



449	{



450		CTarget -= deltaC;



451	}



452



453	yr = (0.5F * CTarget - Cappend)*objPosX2 - 1.5F * LaneW;



454	yl = (0.5F * CTarget + Cappend)*objPosX2 + 1.5F * LaneW;



455



456	if (objPosY > yl || objPosY < yr)



457	{



458		inROIflag = 0;



459	}



460	else



461	{



462		inROIflag = 1;



463	}



464	return inROIflag;



465}

467/*******************************************************************************



468* @name        OutputAdapter_CopyFusionInternalStateToOutput



469* @brief	   This function copy the fusion internal object state into the output definition.



470*			   Only the fusion object with valid oID are copied into the output structure.



471*	           Object without oID should be low priority object.



472* @param[in]   none



473* @param[out]  fusion output info



474* @return      ZONE_RETURN_OK: success



475*              ZONE_RETURN_ERR: fail



476* @see         none



477* @note        none



478******************************************************************************/



479



480sint8 OutputAdapter_CopyFusionInternalStateToOutput(FusionObjOutInfo* pFusOutInfo)



481{



482	FusionSystemInternalInfo*	pFusIntInfo = &gFusionIntInfo;



483



484	// Copy the system info



485	pFusOutInfo->timeStamp = pFusIntInfo->timeStamp;



486	pFusOutInfo->frameNum = pFusIntInfo->lookIndex;



487



488	// Copy the state of each object



489	for (FusionObjInternalStateMap::iterator iterFusion = pFusIntInfo->fusObjIntState.begin(); iterFusion != pFusIntInfo->fusObjIntState.end(); ++iterFusion) {



490		FusionObjInternalState& refFusionInternal = iterFusion->second;



491		uint8 oID = refFusionInternal.oID;



492		if (0 < oID) {



493			// This object has valid oID, so it will be copied into the



494



495			// FUSION obj information



496



497			// used for function safety and kpi



498			if (refFusionInternal.isClusterInside)



499			{



500				RadarClusterMap::iterator iterCluster = pFusIntInfo->clusterDetail.find(refFusionInternal.clusterID);



501				ZONE_Assert(pFusIntInfo->clusterDetail.end() != iterCluster);



502				ZONE_Assert(0 < iterCluster->second.radarID.size());



503				pFusOutInfo->objects[oID - 1].obj_property.visionId = iterCluster->second.radarID.begin()->first;



504			}



505			if (refFusionInternal.isVisInside)



506			{



507				uint32* tempVisID = (uint32*)&pFusOutInfo->objects[oID - 1].obj_information.distanceToleftNearLaneMarking;



508				*tempVisID = refFusionInternal.visID;



509				// used for dump



510				pFusOutInfo->objects[oID - 1].obj_information.distanceToRightNearLaneMarking = static_cast<float32>(refFusionInternal.visID);



511			}



512



513			pFusOutInfo->objects[oID - 1].obj_information.distanceToleftNearLaneMarking = refFusionInternal.distanceToleftNearLaneMarking;



514			pFusOutInfo->objects[oID - 1].obj_information.distanceToRightNearLaneMarking = refFusionInternal.distanceToRightNearLaneMarking;



515			pFusOutInfo->objects[oID - 1].obj_information.motionPatternHistory = refFusionInternal.motionPatternHistory;



516			pFusOutInfo->objects[oID - 1].obj_information.motionPatternCurrent = refFusionInternal.motionPatternCurrent;



517			pFusOutInfo->objects[oID - 1].obj_information.nearestSide = refFusionInternal.nearestSide;



518			pFusOutInfo->objects[oID - 1].obj_information.type = refFusionInternal.object_class;



519			pFusOutInfo->objects[oID - 1].obj_information.height = refFusionInternal.length; // TODO: varify that this height should be length?????



520			pFusOutInfo->objects[oID - 1].obj_information.width = refFusionInternal.width;



521



522			// FUSION obj property



523			pFusOutInfo->objects[oID - 1].obj_property.accelerationStdDev = refFusionInternal.accelerationStdDev;



524			pFusOutInfo->objects[oID - 1].obj_property.headingAngleStdDev = refFusionInternal.headingAngleStdDev;



525			pFusOutInfo->objects[oID - 1].obj_property.latPositionStdDev = refFusionInternal.latPositionStdDev;



526			pFusOutInfo->objects[oID - 1].obj_property.longPositionStdDev = refFusionInternal.longPositionStdDev;



527			pFusOutInfo->objects[oID - 1].obj_property.speedStdDev = refFusionInternal.speedStdDev;



528			//pFusOutInfo->objects[oID - 1].obj_property.cmbbPrimaryConfidence = refFusionInternal.cmbbPrimaryConfidence;



529			//pFusOutInfo->objects[oID - 1].obj_property.cmbbSecondaryConfidence = refFusionInternal.cmbbSecondaryConfidence;



530			pFusOutInfo->objects[oID - 1].obj_property.fcwConfidence = refFusionInternal.fcwConfidence;



531			pFusOutInfo->objects[oID - 1].obj_property.fusionSource = refFusionInternal.fusionSource;
