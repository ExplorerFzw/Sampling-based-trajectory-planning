68/*******************************************************************************



69* @name        Sync_FusionPredict_Run_V



70* @brief



71* @param[in]   none



72* @param[out]  none



73* @return      ZONE_RETURN_OK: success



74*              ZONE_RETURN_ERR: fail



75* @see         none



76* @note        none



77******************************************************************************/



78



79int32_t Sync_FusionPredict_Run_V(void)



80{



81	ZONE_INFO("running to: %s\r\n", __FUNCTION__);



82	Coordinate					trackObjCoordinate;



83	Velocity					trackObjVelocity;



84	Coordinate					predictedCoodinate;



85	Velocity					predictedVelocity;



86	FusionSystemInternalInfo*	pFusIntInfo = &gFusionIntInfo;



87



88	/*recently, sint32 is used here, timestamp overflow is not consider*/



89	sint64 timeDiff		= pFusIntInfo->visionTiemstamp - pFusIntInfo->timeStamp;



90	/*recently, float32 is used here, timestamp overflow is not consider*/



91	float32 fTimeDiff	= timeDiff / FUSION_US_TO_S; // translate us to s; fTimeDiff: unit [s]



92



93	for (FusionObjInternalStateMap::iterator iterFusion = pFusIntInfo->fusObjIntState.begin(); iterFusion != pFusIntInfo->fusObjIntState.end(); ++iterFusion)



94	{



95		FusionObjInternalState& fusObjIntState = iterFusion->second;



96



97		float32 hostYaw		= pFusIntInfo->egoSpeedAndYawrate.EgoYawRateVal * fTimeDiff;



98		float32 f_cos		= cos(hostYaw);



99		float32 f_sin		= sin(hostYaw);



100		float32 longAcc		= fusObjIntState.longAcceleration;



101		float32 lateralAcc	= fusObjIntState.latAcceleration;// + pHostMotion->yawrate.val * pHostMotion->speed.val



102



103		fusObjIntState.longAcceleration = longAcc * f_cos + lateralAcc * f_sin;



104		fusObjIntState.latAcceleration	= lateralAcc * f_cos - longAcc * f_sin;



105



106		/*internal track velocity prediction*/



107		trackObjVelocity.longVel		= fusObjIntState.longVel;



108		trackObjVelocity.lateralVel		= fusObjIntState.latVel;



109		predictedVelocity				= Sync_ObjVelocityPrediction(&trackObjVelocity, pFusIntInfo->egoSpeedAndYawrate.EgoYawRateVal, fTimeDiff);



110		/*internal track coordinate prediction*/



111		trackObjCoordinate.longitudinal = fusObjIntState.longPosn;



112		trackObjCoordinate.lateral		= fusObjIntState.latPosn;



113		predictedCoodinate				= Sync_ObjCoordinatePrediction(&trackObjCoordinate, &predictedVelocity, &(pFusIntInfo->egoSpeedAndYawrate), fTimeDiff);



114		predictedVelocity.longVel		= predictedVelocity.longVel + fusObjIntState.longAcceleration * fTimeDiff;



115		predictedVelocity.lateralVel	= predictedVelocity.lateralVel + fusObjIntState.latAcceleration * fTimeDiff;



116		fusObjIntState.longVel			= predictedVelocity.longVel;



117		fusObjIntState.latVel			= predictedVelocity.lateralVel;



118



119		//Record the predicted value.



120		fusObjIntState.predLongPosn		= predictedCoodinate.longitudinal + 0.5f * fusObjIntState.longAcceleration * fTimeDiff * fTimeDiff;



121		fusObjIntState.predLatPosn		= predictedCoodinate.lateral + 0.5f * fusObjIntState.latAcceleration * fTimeDiff * fTimeDiff;



122



123		fusObjIntState.predVx			= predictedVelocity.longVel;



124		fusObjIntState.predVy			= predictedVelocity.lateralVel;



125		fusObjIntState.predSpeed		= sqrt((predictedVelocity.longVel * predictedVelocity.longVel) + (predictedVelocity.lateralVel * predictedVelocity.lateralVel));



126		fusObjIntState.predHeadingAngle = atan2f(predictedVelocity.lateralVel, predictedVelocity.longVel);



127		fusObjIntState.predAcceleration = fusObjIntState.acceleration;



128



129		fusObjIntState.longPosn			= fusObjIntState.predLongPosn;



130		fusObjIntState.latPosn			= fusObjIntState.predLatPosn;



131	}



132



133	/*update fusion timestamp*/



134	pFusIntInfo->timeStampPrevious		= pFusIntInfo->timeStamp;



135	pFusIntInfo->timeStamp				= pFusIntInfo->visionTiemstamp;



136



137	return ZONE_RETURN_OK;



138}
