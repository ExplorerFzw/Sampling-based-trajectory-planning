68/*******************************************************************************



69* @name        ClusterManager_Group_Run_V



70* @brief



71* @param[in]   none



72* @param[out]  none



73* @return      ZONE_RETURN_OK: success



74*              ZONE_RETURN_ERR: fail



75* @see         none



76* @note        none



77******************************************************************************/



78



79int32_t ClusterManager_Group_Run_V(void)



80{



81	ZONE_INFO("running to: %s\r\n", __FUNCTION__);



82



83	FusionSystemInternalInfo*	pFusIntInfo		= &gFusionIntInfo;



84	FUSION_AlgParameter*		pFusAlgParams	= &gFusionAlgParameter;



85



86	bool flag = false;



87	for (FusionObjInternalStateMap::iterator iterFusion1 = pFusIntInfo->fusObjIntState.begin(); iterFusion1 != pFusIntInfo->fusObjIntState.end();)



88	{



89		FusionObjInternalState& fus_objID1 = iterFusion1->second;



90		flag = false;



91		//stationary obj, no group;vision only track, no group



92		if (fus_objID1.isClusterInside == 1)



93		{



94			RadarClusterMap::iterator clusterIter1 = pFusIntInfo->clusterDetail.find(fus_objID1.clusterID);



95			ZONE_Assert(pFusIntInfo->clusterDetail.end() != clusterIter1);



96			RadarCluster& refClusterDetail1 = clusterIter1->second;



97			if (fus_objID1.isVisInside != 1)



98			{



99				if (refClusterDetail1.objObstacleProb < pFusAlgParams->FUSIONGLOB_OBSROBTHRES)



100				{



101					++iterFusion1;



102					continue;



103				}



104			}



105



106			FusionObjInternalStateMap::iterator iterFusion2 = iterFusion1;



107			++iterFusion2;



108			if (iterFusion2 == pFusIntInfo->fusObjIntState.end())



109			{



110				++iterFusion1;



111				continue;



112			}



113			for (; iterFusion2 != pFusIntInfo->fusObjIntState.end();)



114			{



115				FusionObjInternalState& fus_objID2 = iterFusion2->second;



116				if (fus_objID2.isClusterInside == 1)



117				{



118					RadarClusterMap::iterator clusterIter2 = pFusIntInfo->clusterDetail.find(fus_objID2.clusterID);



119					ZONE_Assert(pFusIntInfo->clusterDetail.end() != clusterIter2);



120					RadarCluster& refClusterDetail2 = clusterIter2->second;



121					if (fus_objID2.isVisInside != 1)



122					{



123						if (refClusterDetail2.objObstacleProb < pFusAlgParams->FUSIONGLOB_OBSROBTHRES)



124						{



125							++iterFusion2;



126							continue;



127						}



128					}



129					if ((fus_objID1.isVisInside == 1) && (fus_objID2.isVisInside == 1))



130					{



131						//both track is VisInside,no group



132						++iterFusion2;



133						continue;



134					}



135					else



136					{



137						float32 deltaHeading = fabs(fus_objID1.headingAngle - fus_objID2.headingAngle);



138						float32 deltaV = sqrt((fus_objID1.longVel - fus_objID2.longVel) * (fus_objID1.longVel - fus_objID2.longVel)



139							+ (fus_objID1.latVel - fus_objID2.latVel) * (fus_objID1.latVel - fus_objID2.latVel));



140



141						if (deltaHeading > GROUP_HEADING || deltaV > GROUP_V)



142						{



143							++iterFusion2;



144							continue;



145						}



146						else



147						{



148							//cal coordinate of bbox2 center in bbox1 coordinate system



149							float dx = fus_objID2.obj_boundary.Xcenter - fus_objID1.obj_boundary.Xcenter;



150							float dy = fus_objID2.obj_boundary.Ycenter - fus_objID1.obj_boundary.Ycenter;



151							float x2 = dx*cosf(fus_objID1.headingAngle) + dy*sinf(fus_objID1.headingAngle);



152							float y2 = -dx*sinf(fus_objID1.headingAngle) + dy*cosf(fus_objID1.headingAngle);



153							float f_groupPX = ZONE_MAX(GROUP_PX, ZONE_MAX(0.8F*fus_objID1.length, 0.8F*fus_objID2.length));



154							float f_groupPY = ZONE_MAX(GROUP_PY, ZONE_MAX(0.5F*fus_objID1.width, 0.5F*fus_objID2.width));



155



156							if (fabs(y2) > f_groupPY || fabs(x2) > f_groupPX)



157							{



158								++iterFusion2;



159								continue;



160							}



161							else



162							{



163								//group,update fusionID and other para



164								if (fus_objID2.isVisInside == 1)



165								{



166									//remain fusionID2,delete fusionID1



167									float ratiotemp = fus_objID2.latPosn / fus_objID2.longPosn;



168									fus_objID2.longPosn = ZONE_MIN(fus_objID2.longPosn, fus_objID1.longPosn);



169									fus_objID2.latPosn = ratiotemp * fus_objID2.longPosn;



170									fus_objID2.longVel = 0.5F * (fus_objID2.longVel + fus_objID1.longVel);



171									fus_objID2.latVel = 0.5F * (fus_objID2.latVel + fus_objID1.latVel);



172									float maxL = ZONE_MAX(fus_objID1.length*0.5F, x2 + fus_objID2.length*0.5F);



173									float minL = ZONE_MIN(-fus_objID1.length*0.5F, x2 - fus_objID2.length*0.5F);



174									float maxW = ZONE_MAX(fus_objID1.width*0.5F, y2 + fus_objID2.width*0.5F);



175									float minW = ZONE_MIN(-fus_objID1.width*0.5F, y2 - fus_objID2.width*0.5F);



176									fus_objID2.length = maxL - minL;



177									fus_objID2.width = maxW - minW;



178									float centerXtemp = minL + 0.5F * fus_objID2.length;



179									float centerYtemp = minW + 0.5F * fus_objID2.width;



180									fus_objID2.obj_boundary.Xcenter = centerXtemp * cosf(fus_objID1.headingAngle) - centerYtemp * sinf(fus_objID1.headingAngle) + fus_objID1.obj_boundary.Xcenter;



181									fus_objID2.obj_boundary.Ycenter = centerXtemp * sinf(fus_objID1.headingAngle) + centerYtemp * cosf(fus_objID1.headingAngle) + fus_objID1.obj_boundary.Ycenter;



182									fus_objID2.headingAngle = 0.5F * (fus_objID1.headingAngle + fus_objID2.headingAngle);



183



184									//merge two clusters					



185									refClusterDetail2.numOfRadar = refClusterDetail2.numOfRadar + refClusterDetail1.numOfRadar;



186									for (radarIdMap::iterator iterRadarID = refClusterDetail1.radarID.begin(); iterRadarID != refClusterDetail1.radarID.end(); ++iterRadarID)



187									{



188										refClusterDetail2.radarID.insert(std::make_pair(iterRadarID->first, 0));



189										//pFusionSystemInternalInfo->radarIntDetail.at(iterRadarID->first).ClusterID = refClusterDetail2.ID;



190										RadarTrackDetailMap::iterator radarIter = pFusIntInfo->radarIntDetail.find(iterRadarID->first);



191										ZONE_Assert(pFusIntInfo->radarIntDetail.end() != radarIter);



192										RadarTrackDetail& refRadarTemp = radarIter->second;



193										refRadarTemp.ClusterID = refClusterDetail2.ID;



194



195									}



196									refClusterDetail1.numOfRadar = 0;



197									refClusterDetail1.radarID.clear();



198



199									//delete the  fusionID which is grouped



200									iterFusion1 = Del_Track(pFusIntInfo, iterFusion1);



201									flag = true;



202									break;



203								}



204								else if (fus_objID1.isVisInside == 1)



205								{



206									//remain fusionID1,delete fusionID2



207									float ratiotemp		= fus_objID1.latPosn / fus_objID1.longPosn;



208									fus_objID1.longPosn = ZONE_MIN(fus_objID1.longPosn, fus_objID2.longPosn);



209									fus_objID1.latPosn	= ratiotemp * fus_objID1.longPosn;



210									fus_objID1.longVel	= 0.5F * (fus_objID2.longVel + fus_objID1.longVel);



211									fus_objID1.latVel	= 0.5F * (fus_objID2.latVel + fus_objID1.latVel);



212									float maxL = ZONE_MAX(fus_objID1.length*0.5F, x2 + fus_objID2.length*0.5F);



213									float minL = ZONE_MIN(-fus_objID1.length*0.5F, x2 - fus_objID2.length*0.5F);



214									float maxW = ZONE_MAX(fus_objID1.width*0.5F, y2 + fus_objID2.width*0.5F);



215									float minW = ZONE_MIN(-fus_objID1.width*0.5F, y2 - fus_objID2.width*0.5F);



216									fus_objID1.length	= maxL - minL;



217									fus_objID1.width	= maxW - minW;



218									float centerXtemp	= minL + 0.5F * fus_objID1.length;



219									float centerYtemp	= minW + 0.5F * fus_objID1.width;



220									fus_objID1.obj_boundary.Xcenter = centerXtemp * cosf(fus_objID1.headingAngle) - centerYtemp * sinf(fus_objID1.headingAngle) + fus_objID1.obj_boundary.Xcenter;



221									fus_objID1.obj_boundary.Ycenter = centerXtemp * sinf(fus_objID1.headingAngle) + centerYtemp * cosf(fus_objID1.headingAngle) + fus_objID1.obj_boundary.Ycenter;



222									fus_objID1.headingAngle			= 0.5F * (fus_objID1.headingAngle + fus_objID2.headingAngle);



223



224									//merge two clusters



225									refClusterDetail1.numOfRadar = refClusterDetail2.numOfRadar + refClusterDetail1.numOfRadar;



226									for (radarIdMap::iterator iterRadarID = refClusterDetail2.radarID.begin(); iterRadarID != refClusterDetail2.radarID.end(); ++iterRadarID)



227									{



228										refClusterDetail1.radarID.insert(std::make_pair(iterRadarID->first, 0));



229										//pFusionSystemInternalInfo->radarIntDetail.at(iterRadarID->first).ClusterID = refClusterDetail1.ID;



230										RadarTrackDetailMap::iterator radarIter = pFusIntInfo->radarIntDetail.find(iterRadarID->first);



231										ZONE_Assert(pFusIntInfo->radarIntDetail.end() != radarIter);



232										RadarTrackDetail& refRadarTemp = radarIter->second;



233										refRadarTemp.ClusterID = refClusterDetail1.ID;



234									}



235									refClusterDetail2.numOfRadar = 0;



236									refClusterDetail2.radarID.clear();



237



238									iterFusion2 = Del_Track(pFusIntInfo, iterFusion2);



239								}



240								else



241								{



242									if (fus_objID1.visionConfirmCnt > fus_objID2.visionConfirmCnt)



243									{



244										//remain fusionID1,delete fusionID2



245										fus_objID1.longPosn = ZONE_MIN(fus_objID1.longPosn, fus_objID2.longPosn);



246										fus_objID1.latPosn = 0.5F * (fus_objID2.latPosn + fus_objID1.latPosn);



247										fus_objID1.longVel = 0.5F * (fus_objID2.longVel + fus_objID1.longVel);



248										fus_objID1.latVel = 0.5F * (fus_objID2.latVel + fus_objID1.latVel);



249



250										float maxL = ZONE_MAX(fus_objID1.length*0.5F, x2 + fus_objID2.length*0.5F);



251										float minL = ZONE_MIN(-fus_objID1.length*0.5F, x2 - fus_objID2.length*0.5F);



252										float maxW = ZONE_MAX(fus_objID1.width*0.5F, y2 + fus_objID2.width*0.5F);



253										float minW = ZONE_MIN(-fus_objID1.width*0.5F, y2 - fus_objID2.width*0.5F);



254										fus_objID1.length = maxL - minL;



255										fus_objID1.width = maxW - minW;



256										float centerXtemp = minL + 0.5F * fus_objID1.length;



257										float centerYtemp = minW + 0.5F * fus_objID1.width;



258										fus_objID1.obj_boundary.Xcenter = centerXtemp * cosf(fus_objID1.headingAngle) - centerYtemp * sinf(fus_objID1.headingAngle) + fus_objID1.obj_boundary.Xcenter;



259										fus_objID1.obj_boundary.Ycenter = centerXtemp * sinf(fus_objID1.headingAngle) + centerYtemp * cosf(fus_objID1.headingAngle) + fus_objID1.obj_boundary.Ycenter;



260										fus_objID1.headingAngle = 0.5F * (fus_objID1.headingAngle + fus_objID2.headingAngle);



261



262										//merge two clusters



263										refClusterDetail1.numOfRadar = refClusterDetail2.numOfRadar + refClusterDetail1.numOfRadar;



264										for (radarIdMap::iterator iterRadarID = refClusterDetail2.radarID.begin(); iterRadarID != refClusterDetail2.radarID.end(); ++iterRadarID)



265										{



266											refClusterDetail1.radarID.insert(std::make_pair(iterRadarID->first, 0));



267											//pFusionSystemInternalInfo->radarIntDetail.at(iterRadarID->first).ClusterID = refClusterDetail1.ID;



268											RadarTrackDetailMap::iterator radarIter = pFusIntInfo->radarIntDetail.find(iterRadarID->first);



269											ZONE_Assert(pFusIntInfo->radarIntDetail.end() != radarIter);



270											RadarTrackDetail& refRadarTemp = radarIter->second;



271											refRadarTemp.ClusterID = refClusterDetail1.ID;



272										}



273										refClusterDetail2.numOfRadar = 0;



274										refClusterDetail2.radarID.clear();



275



276										iterFusion2 = Del_Track(pFusIntInfo, iterFusion2);



277									}



278									else



279									{



280										//remain fusionID1,delete fusionID2



281										fus_objID2.longPosn = ZONE_MIN(fus_objID2.longPosn, fus_objID1.longPosn);



282										fus_objID2.latPosn = 0.5F * (fus_objID1.latPosn + fus_objID2.latPosn);



283										fus_objID2.longVel = 0.5F * (fus_objID1.longVel + fus_objID2.longVel);



284										fus_objID2.latVel = 0.5F * (fus_objID1.latVel + fus_objID2.latVel);



285



286										float maxL = ZONE_MAX(fus_objID1.length*0.5F, x2 + fus_objID2.length*0.5F);



287										float minL = ZONE_MIN(-fus_objID1.length*0.5F, x2 - fus_objID2.length*0.5F);



288										float maxW = ZONE_MAX(fus_objID1.width*0.5F, y2 + fus_objID2.width*0.5F);



289										float minW = ZONE_MIN(-fus_objID1.width*0.5F, y2 - fus_objID2.width*0.5F);



290										fus_objID2.length = maxL - minL;



291										fus_objID2.width = maxW - minW;



292										float centerXtemp = minL + 0.5F * fus_objID2.length;



293										float centerYtemp = minW + 0.5F * fus_objID2.width;



294										fus_objID2.obj_boundary.Xcenter = centerXtemp * cosf(fus_objID2.headingAngle) - centerYtemp * sinf(fus_objID2.headingAngle) + fus_objID2.obj_boundary.Xcenter;



295										fus_objID2.obj_boundary.Ycenter = centerXtemp * sinf(fus_objID2.headingAngle) + centerYtemp * cosf(fus_objID2.headingAngle) + fus_objID2.obj_boundary.Ycenter;



296										fus_objID2.headingAngle = 0.5F * (fus_objID2.headingAngle + fus_objID1.headingAngle);



297



298										//merge two clusters



299										refClusterDetail2.numOfRadar = refClusterDetail1.numOfRadar + refClusterDetail2.numOfRadar;



300										for (radarIdMap::iterator iterRadarID = refClusterDetail1.radarID.begin(); iterRadarID != refClusterDetail1.radarID.end(); ++iterRadarID)



301										{



302											refClusterDetail2.radarID.insert(std::make_pair(iterRadarID->first, 0));



303											//pFusionSystemInternalInfo->radarIntDetail.at(iterRadarID->first).ClusterID = refClusterDetail1.ID;



304											RadarTrackDetailMap::iterator radarIter = pFusIntInfo->radarIntDetail.find(iterRadarID->first);



305											ZONE_Assert(pFusIntInfo->radarIntDetail.end() != radarIter);



306											RadarTrackDetail& refRadarTemp = radarIter->second;



307											refRadarTemp.ClusterID = refClusterDetail2.ID;



308										}



309										refClusterDetail1.numOfRadar = 0;



310										refClusterDetail1.radarID.clear();



311



312										//delete the  fusionID which is grouped



313										iterFusion1 = Del_Track(pFusIntInfo, iterFusion1);



314										flag = true;



315										break;



316									}



317								}



318								//delete the  fusionID which is grouped



319								//iterFusion2 = Del_Track(pFusionSystemInternalInfo, iterFusion2);



320							}



321						}



322					}



323				}



324				else



325				{



326					//no group



327					++iterFusion2;



328				}



329			}



330		}



331		if (flag == false)



332		{



333			++iterFusion1;



334		}



335	}



336	return ZONE_RETURN_OK;
