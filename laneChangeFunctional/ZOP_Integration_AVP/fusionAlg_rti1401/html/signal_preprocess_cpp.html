359/*******************************************************************************



360* @name        SignalPreprocess_RadarDoubleBounceDet



361* @brief       Radar double bounce detector



362* @param[in]   pRadarInfo: radar input object list



363* @param[out]  validity[]: radar internal object



364* @return      ZONE_RETURN_OK: success



365*              ZONE_RETURN_ERR: fail



366* @see         none



367* @note        none



368******************************************************************************/



369



370bool xPosSort(std::pair<int, const RadarObj*> a, std::pair<int, const RadarObj*> b)



371{



372	return a.second->objXPos < b.second->objXPos;



373}



374



375inline sint8 SignalPreprocess_RadarDoubleBounceDet(const CanRadarObjList* pRadarInfo, uint8 validity[])



376{



377	const float32 radarLong = RADAR_MOUNT_LONG_POSN;



378	std::vector<std::pair<int, const RadarObj*> > radarData;



379	radarData.reserve(UTILS_RADAR_MAX_OBJ_NUM);



380



381	for (uint8 iCnt = 0; iCnt < UTILS_RADAR_MAX_OBJ_NUM; ++iCnt)



382	{



383		const RadarObj* pRadarObj = &pRadarInfo->radarObj[iCnt];



384		if (FUSION_OBJECT_VAILID == pRadarObj->objValidFlag)



385		{



386			if ((pRadarObj->objXPos >= THRESHOS_XPOS_NEAR) && (fabs(pRadarObj->objYPos) <= THRESHOS_YPOS)) // do not detect double bounce if X/Y position if out of range



387			{



388				radarData.push_back(std::pair<int, const RadarObj*>(iCnt, pRadarObj));



389			}



390		}



391	}



392	if (2 > radarData.size())



393	{



394		return 0;



395	}



396



397	std::sort(radarData.begin(), radarData.end(), xPosSort);



398	for (std::vector<std::pair<int, const RadarObj*> >::const_iterator itera = radarData.begin(); itera != radarData.end(); ++itera)



399	{



400		const RadarObj* pRadarObjA = itera->second;



401		if (pRadarObjA->objXPos > 40.0F) // if the radar object xPos is farther than 40m it will not have double bounce in principle



402		{



403			break;



404		}



405		if (validity[itera->first] == FUSION_OBJECT_INVALID) // if the radar object already mark as a double bounce object, do not need to find its double bounce object again



406		{



407			continue;



408		}



409		for (std::vector<std::pair<int, const RadarObj*> >::const_iterator iterb = itera + 1; iterb != radarData.end(); ++iterb)



410		{



411			const RadarObj* pRadarObjB = iterb->second;



412



413			if (THRESHOS_XPOS_FAR < pRadarObjB->objXPos)



414			{



415				break;



416			}



417



418			float32 xPosComp = (pRadarObjB->objXPos - radarLong) / (pRadarObjA->objXPos - radarLong);



419			float32 xVelComp = 0.0F;



420			uint8 isVxPass = 0;



421			if (fabs(pRadarObjA->objXVelRel) > 0.1F)



422			{



423				xVelComp = pRadarObjB->objXVelRel / pRadarObjA->objXVelRel;



424			}



425			else



426			{



427				if (fabs(pRadarObjB->objXVelRel - pRadarObjA->objXVelRel) <= 0.12F)



428				{



429					isVxPass = 1;



430				}



431			}



432			float32 dltYPos = fabs(pRadarObjB->objYPos - pRadarObjA->objYPos);



433			/*find double bounce radar object*/



434			if ((xPosComp >= 1.9F) && (xPosComp <= 2.1F)



435				&& (((xVelComp >= 1.9F) && (xVelComp <= 2.1F)) || (isVxPass == 1))



436				&& dltYPos < 1.0F)



437			{



438				// must be a double bounce radar object



439				validity[iterb->first] = 0;



440				break;



441			}



442			else if ((xPosComp >= 1.5F) && (xPosComp <= 2.5F)



443				&& (((xVelComp >= 1.5F) && (xVelComp <= 2.5F)) || (isVxPass == 1))



444				&& (dltYPos < 1.3F))



445			{



446				if (pRadarObjB->objExstProb <= RELIABLE_EXISTPROB)



447				{



448					validity[iterb->first] = 0;



449					break;



450				}



451			}



452		}



453	}



454



455	for (std::vector<std::pair<int, const RadarObj*> >::const_iterator iterRa = radarData.begin(); iterRa != radarData.end(); ++iterRa)



456	{



457		const RadarObj* pRadarObj = iterRa->second;



458		if ((RADARSTATUS_MOVING == pRadarObj->objMotionPattern)



459			&& (2.0F > fabs(pRadarObj->objYPos))



460			&& (RELIABLE_EXISTPROB <= pRadarObj->objExstProb))



461		{



462			for (std::vector<std::pair<int, const RadarObj*> >::const_iterator iterRaS = ++iterRa; iterRaS != radarData.end(); ++iterRaS)



463			{



464				if ((0 != validity[iterRaS->first]) && (2.0F > fabs(iterRaS->second->objYPos)))



465				{



466					validity[iterRaS->first] = 2;



467				}



468			}



469			break;



470		}



471



472	}



473



474	return ZONE_RETURN_OK;



475}

479* @brief



480* @param[in]   originTimestamp: original sensor timestamp

488inline uint64 SignalPreprocess_TimestampCompensation(uint64 originTimestamp, uint64 latency)

494/*******************************************************************************



495* @name        SignalPreprocess_CompareTTC



496* @brief



497* @param[in]   refVisTrackDetail: vision track data



498*			   pHostMotion: the vehicle data (yawRate and speed) of the current time



499* @param[out]  none



500* @return      ZONE_RETURN_OK: success



501*              ZONE_RETURN_ERR: fail



502* @see         none



503* @note        none



504******************************************************************************/



505



506inline sint8 SignalPreprocess_CompareTTC(VisTrackDetail& refVisTrackDetail, const EgoParam* pHostMotion)



507{



508	float32 vRel			= pHostMotion->EgoSpeedVal - refVisTrackDetail.obj_velocity.Vx;



509	float32 TTCVision		= refVisTrackDetail.obj_TTC;



510	float32 TTCCal			= 0.0F;



511	const float32 dTTCMax	= 1.5F;



512



513	if (vRel > 1.0e-6F)



514	{



515		TTCCal = refVisTrackDetail.obj_coordinate.longitudinal / vRel;



516		if (TTCVision > 3.5F)



517		{



518			if (TTCCal < 3.5F - dTTCMax)



519			{



520				refVisTrackDetail.obj_coordinate.longitudinalVariance = 10.0F;



521			}



522		}



523		else



524		{



525			if ((TTCCal < TTCVision - dTTCMax) || (TTCCal > TTCVision + dTTCMax))



526			{



527				refVisTrackDetail.obj_coordinate.longitudinalVariance = 10.0F;



528			}



529		}



530	}



531	return ZONE_RETURN_OK;



532}

546inline sint8 SignalPreprocess_RadarRelToAbsolute(const RadarObj* pRadarObj, const EgoParam* pHostMotion, RadarTrackDetail* pRadarTrackDetail)



547{



548	float32 yawRate = pHostMotion->EgoYawRateVal;



549	float32 hostXVel = pHostMotion->EgoSpeedVal; // pHostMotion->EgoSpeedVal; // host longitudinal velocity;



550	float32 hostYVel = 0.0F; // host lateral velocity



551



552	///*calculate radar rotation vector based on host yaw rate and radar position*/



553	//Vector3d hostYawRate(0.0F, 0.0F, pEgoMotion->yawrate.val);



554	//Vector3d radarCoordinate(RADAR_MOUNT_LONG_POSN, RADAR_MOUNT_LAT_POSN, 0.0F);



555	//Vector3d radarRotationRate = hostYawRate.cross(radarCoordinate);



556



557	///*calculate radar velocity based on radar rotation vector*/



558	//float32 radarXVel1 = hostXVel + radarRotationRate(0);



559	//float32 radarYVel1 = hostYVel + radarRotationRate(1);



560



561	///*directly cal without Eigen*/



562	float32 Robj = sqrt(powf(pRadarObj->objXPos, 2) + powf(pRadarObj->objYPos, 2));



563	float32 thetaObj = atan2f(pRadarObj->objYPos, pRadarObj->objXPos);



564	float32 radarXVel = hostXVel - yawRate * Robj *  sinf(thetaObj);



565	float32 radarYVel = hostYVel + yawRate * Robj *  cosf(thetaObj);



566



567	/*calculate radar object absolute velocity based on radar velocity*/



568	pRadarTrackDetail->objXVelAbs = pRadarObj->objXVelRel + radarXVel;



569	pRadarTrackDetail->objYVelAbs = pRadarObj->objYVelRel + radarYVel;



570



571	/*maintain the radar object acceleration*/



572	pRadarTrackDetail->objXAccAbs = pRadarObj->objXAccRel;



573



574	return ZONE_RETURN_OK;
