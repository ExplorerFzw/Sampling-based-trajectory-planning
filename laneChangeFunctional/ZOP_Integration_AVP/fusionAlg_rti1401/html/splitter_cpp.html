68/*******************************************************************************



69* @name        ClusterManager_Splitter_Run_V



70* @brief



71* @param[in]   none



72* @param[out]  none



73* @return      ZONE_RETURN_OK: success



74*              ZONE_RETURN_ERR: fail



75* @see         none



76* @note        none



77******************************************************************************/



78



79int32_t ClusterManager_Splitter_Run_V(void)



80{



81	ZONE_INFO("running to: %s\r\n", __FUNCTION__);



82



83	float32_t SPLIT_X1, SPLIT_X2, SPLIT_Y1, SPLIT_Y2;



84	float32_t fusionX, fusionY, fusionVX, fusionVY;



85	bool flagP1, flagP2, flagVX, flagVY;



86	float32_t f_distance;



87	uint8_t splitRadarID[UTILS_RADAR_MAX_OBJ_NUM];



88	int32_t countSplitRadarID = 0;



89



90	FusionSystemInternalInfo*	pFusIntInfo = &gFusionIntInfo;



91



92	for (FusionObjInternalStateMap::iterator iterFusion = pFusIntInfo->fusObjIntState.begin(); iterFusion != pFusIntInfo->fusObjIntState.end(); ++iterFusion)



93	{



94		FusionObjInternalState& refFusion = iterFusion->second;



95		if (refFusion.isClusterInside)



96		{



97			//RadarCluster& refCluster = pFusionSystemInternalInfo->clusterDetail.at(refFusion.clusterID);



98			RadarClusterMap::iterator clusterIter = pFusIntInfo->clusterDetail.find(refFusion.clusterID);



99			ZONE_Assert(pFusIntInfo->clusterDetail.end() != clusterIter);



100			RadarCluster& refCluster = clusterIter->second;



101



102			if (refCluster.numOfRadar > 1)



103			{



104				fusionX = refFusion.longPosn;



105				fusionY = refFusion.latPosn;



106				fusionVX = refFusion.longVel;



107				fusionVY = refFusion.latVel;



108				SPLIT_X1 = refFusion.length*0.8F*1.1F;



109				SPLIT_X2 = refFusion.length*0.8F*1.3F;



110				SPLIT_Y1 = refFusion.width*0.8F*1.1F;



111				SPLIT_Y2 = refFusion.width*0.8F*1.2F;



112



113				int countRadar = refCluster.numOfRadar;



114				for (radarIdMap::iterator iterRadarId = refCluster.radarID.begin(); iterRadarId != refCluster.radarID.end(); ++iterRadarId)



115				{



116					if (countRadar <= 1)



117					{



118						break;



119					}



120					//RadarTrackDetail& refRadarTrackDetail = pFusionSystemInternalInfo->radarIntDetail.at(iterRadarId->first);



121					RadarTrackDetailMap::iterator radarIter = pFusIntInfo->radarIntDetail.find(iterRadarId->first);



122					ZONE_Assert(pFusIntInfo->radarIntDetail.end() != radarIter);



123					RadarTrackDetail& refRadarTrackDetail = radarIter->second;



124



125					flagVX = fabs(refRadarTrackDetail.objXVelAbs - fusionVX) < SPLIT_VX;



126					flagVY = fabs(refRadarTrackDetail.objYVelAbs - fusionVY) < SPLIT_VY;



127



128					if (flagVX && flagVY)



129					{



130						//cal coordinate of radarPoint in bbox coordinate system



131						float dx = refRadarTrackDetail.objXPos - refFusion.obj_boundary.Xcenter;



132						float dy = refRadarTrackDetail.objYPos - refFusion.obj_boundary.Ycenter;



133						float x2 = dx*cosf(refFusion.headingAngle) + dy*sinf(refFusion.headingAngle);



134						float y2 = -dx*sinf(refFusion.headingAngle) + dy*cosf(refFusion.headingAngle);



135



136						flagP1 = (fusionVX < SPLIT_V) && (x2 < SPLIT_X1) && (y2 < SPLIT_Y1);



137						flagP2 = (fusionVX >= SPLIT_V) && (x2 < SPLIT_X2) && (y2 < SPLIT_Y2);



138



139						if (flagP1 || flagP2)



140						{



141							iterRadarId->second = 0;



142						}



143						else



144						{



145							iterRadarId->second++;



146						}



147					}



148					else



149					{



150						iterRadarId->second++;



151					}



152					//d-velocity or distance is larger than threshold,split it



153					if (iterRadarId->second >= 3)



154					{



155						splitRadarID[countSplitRadarID] = refRadarTrackDetail.objID;



156						++countSplitRadarID;



157



158						countRadar--;



159					}



160				}



161			}



162		}



163	}



164	for (int icount = 0; icount < countSplitRadarID; icount++)



165	{



166		uint8  radarID = splitRadarID[icount];



167		//RadarTrackDetail& refRadarTrackDetail = pFusionSystemInternalInfo->radarIntDetail.at(radarID);



168		RadarTrackDetailMap::iterator radarIter = pFusIntInfo->radarIntDetail.find(radarID);



169		ZONE_Assert(pFusIntInfo->radarIntDetail.end() != radarIter);



170		RadarTrackDetail& refRadarTrackDetail = radarIter->second;



171



172		delRadarInCluster(refRadarTrackDetail, *pFusIntInfo);



173



174		putRadarInNewCluster(refRadarTrackDetail, pFusIntInfo);



175	}



176



177	return ZONE_RETURN_OK;



178}
